3from __future__ import annotations
4
5import numpy as np
6from scipy.stats import t
7
8
9def simulate_gaussian_linear(
10    n: int,
11    d: int,
12    beta: np.ndarray,
13    sigma: float,
14    seed: int | None = None,
15) -> tuple[np.ndarray, np.ndarray]:
16    """Simulate data attached to graph node `model.gaussian_observation`."""
17    rng = np.random.default_rng(seed)
18    X = rng.normal(size=(n, d))
19    noise = rng.normal(scale=sigma, size=n)
20    y = X @ np.asarray(beta, dtype=float) + noise
21    return X, y

24def simulate_student_t_linear(
25    n: int,
26    d: int,
27    beta: np.ndarray,
28    sigma: float,
29    nu: float,
30    seed: int | None = None,
31) -> tuple[np.ndarray, np.ndarray]:
32    """Simulate data attached to graph node `model.student_t_noise`."""
33    rng = np.random.default_rng(seed)
34    X = rng.normal(size=(n, d))
35    noise = t.rvs(df=nu, loc=0.0, scale=sigma, size=n, random_state=rng)
36    y = X @ np.asarray(beta, dtype=float) + np.asarray(noise, dtype=float)
37    return X, y
