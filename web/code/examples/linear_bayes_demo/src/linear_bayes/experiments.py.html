1"""Experiment helpers attached to experiment and validation graph nodes."""
2
3from __future__ import annotations
4
5from typing import Any
6
7import numpy as np
8
9from linear_bayes.estimators import beta_map_closed_form
10from linear_bayes.simulate import simulate_gaussian_linear
11
12
13def run_recovery_experiment(
14    n_values: list[int],
15    d: int,
16    sigma: float,
17    tau: float,
18    beta_true: np.ndarray,
19    seeds: list[int],
20) -> dict[str, Any]:
21    """Run synthetic recovery for graph node `experiment.gaussian_recovery`."""
22    beta_true = np.asarray(beta_true, dtype=float)
23    if beta_true.shape != (d,):
24        raise ValueError(f"beta_true must have shape ({d},)")
25    if not n_values:
26        raise ValueError("n_values must not be empty")
27    if not seeds:
28        raise ValueError("seeds must not be empty")
29
30    raw_errors: dict[str, list[float]] = {}
31    mean_errors: list[float] = []
32    std_errors: list[float] = []
33
34    for n in n_values:
35        errors = []
36        for seed in seeds:
37            X, y = simulate_gaussian_linear(n=n, d=d, beta=beta_true, sigma=sigma, seed=seed)
38            beta_hat = beta_map_closed_form(X=X, y=y, sigma=sigma, tau=tau)
39            errors.append(float(np.linalg.norm(beta_hat - beta_true)))
40        raw_errors[str(n)] = errors
41        mean_errors.append(float(np.mean(errors)))
42        std_errors.append(float(np.std(errors, ddof=0)))
43
44    return {
45        "n_values": [int(n) for n in n_values],
46        "mean_errors": mean_errors,
47        "std_errors": std_errors,
48        "raw_errors": raw_errors,
49        "beta_true": beta_true.tolist(),
50        "sigma": float(sigma),
51        "tau": float(tau),
52        "seeds": [int(seed) for seed in seeds],
53    }
