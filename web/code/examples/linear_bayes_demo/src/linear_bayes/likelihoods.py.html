3from __future__ import annotations
4
5import numpy as np
6from scipy.stats import norm, t
7
8
9def gaussian_log_likelihood(
10    X: np.ndarray,
11    y: np.ndarray,
12    beta: np.ndarray,
13    sigma: float,
14) -> float:
15    """Return Gaussian log likelihood for graph node `likelihood.gaussian`."""
16    X = np.asarray(X, dtype=float)
17    y = np.asarray(y, dtype=float)
18    beta = np.asarray(beta, dtype=float)
19    residual = y - X @ beta
20    return float(np.sum(norm.logpdf(residual, loc=0.0, scale=sigma)))

22
23def student_t_log_likelihood(
24    X: np.ndarray,
25    y: np.ndarray,
26    beta: np.ndarray,
27    sigma: float,
28    nu: float,
29) -> float:
30    """Return Student-t log likelihood for graph node `likelihood.student_t`."""
31    X = np.asarray(X, dtype=float)
32    y = np.asarray(y, dtype=float)
33    beta = np.asarray(beta, dtype=float)
34    residual = y - X @ beta
35    return float(np.sum(t.logpdf(residual / sigma, df=nu) - np.log(sigma)))
