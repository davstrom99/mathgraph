examples/linear_bayes_demo/experiments/run_gaussian_recovery.py

1"""Run the Gaussian synthetic recovery experiment.
2
3Attached graph node: `experiment.gaussian_recovery`.
4Declared outputs:
5- examples/linear_bayes_demo/results/gaussian_recovery_summary.json
6- examples/linear_bayes_demo/results/gaussian_recovery_error.png
7"""
8
9from __future__ import annotations
10
11import json
12from pathlib import Path
13
14import matplotlib.pyplot as plt
15import numpy as np
16
17from linear_bayes.experiments import run_recovery_experiment
18
19
20EXAMPLE_ROOT = Path(__file__).resolve().parent.parent
21BETA_TRUE = np.array([1.0, -0.5, 0.25])
22N_VALUES = [20, 50, 100, 250, 500]
23SEEDS = list(range(25))
24SIGMA = 1.0
25TAU = 10.0
26SUMMARY_PATH = EXAMPLE_ROOT / "results" / "gaussian_recovery_summary.json"
27PLOT_PATH = EXAMPLE_ROOT / "results" / "gaussian_recovery_error.png"
28
29
30def main(
31    summary_path: Path = SUMMARY_PATH,
32    plot_path: Path = PLOT_PATH,
33) -> None:
34    """Run and save outputs for graph node `experiment.gaussian_recovery`."""
35    result = run_recovery_experiment(
36        n_values=N_VALUES,
37        d=BETA_TRUE.size,
38        sigma=SIGMA,
39        tau=TAU,
40        beta_true=BETA_TRUE,
41        seeds=SEEDS,
42    )
43
44    summary_path.parent.mkdir(parents=True, exist_ok=True)
45    plot_path.parent.mkdir(parents=True, exist_ok=True)
46
47    summary_path.write_text(json.dumps(result, indent=2), encoding="utf-8")
48    _save_error_plot(result, plot_path)
49
50    first_error = result["mean_errors"][0]
51    last_error = result["mean_errors"][-1]
52    print(f"Wrote {summary_path}")
53    print(f"Wrote {plot_path}")
54    print(f"Mean error: n={N_VALUES[0]} -> {first_error:.4f}, n={N_VALUES[-1]} -> {last_error:.4f}")
55
56
57def _save_error_plot(result: dict, plot_path: Path) -> None:
58    n_values = np.asarray(result["n_values"], dtype=float)
59    mean_errors = np.asarray(result["mean_errors"], dtype=float)
60    std_errors = np.asarray(result["std_errors"], dtype=float)
61
62    fig, ax = plt.subplots(figsize=(7.0, 4.5), constrained_layout=True)
63    ax.errorbar(
64        n_values,
65        mean_errors,
66        yerr=std_errors,
67        marker="o",
68        linewidth=2.0,
69        capsize=4,
70        color="#1f77b4",
71        ecolor="#8fb9dc",
72    )
73    ax.set_xscale("log")
74    ax.set_xlabel("sample size n")
75    ax.set_ylabel(r"$\|\hat{\beta}_{MAP} - \beta_0\|_2$")
76    ax.set_title("Gaussian synthetic recovery")
77    ax.grid(True, which="both", alpha=0.3)
78    fig.savefig(plot_path, dpi=160)
79    plt.close(fig)
80
81
82if __name__ == "__main__":
83    main()
