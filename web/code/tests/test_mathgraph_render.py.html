92def test_tex_converter_renders_align_starred_sections_and_paragraphs():
93    source = r"""\section*{Mathematical objects}
94\begin{align}
95\boldsymbol{\alpha}&=(b_1,c_1,\ldots,b_M,c_M)\in(-1,1)^M\times(-1,0)^M\\
96\mathbf{s}[n]&\in\mathbb{R}^{2M}\\
97p(\mathbf{s}[n]&\mid r[0{:}n])
98\end{align}
99\par Office audio dataset.
100\par Numerical and end-to-end validation suite.
101"""
102
103    rendered = _tex_to_html(source)
104
105    assert "<h2>Mathematical objects</h2>" in rendered
106    assert r"\[\begin{aligned}" in rendered
107    assert r"\boldsymbol{\alpha}" in rendered
108    assert r"\end{aligned}\]" in rendered
109    assert "<p>Office audio dataset.</p>" in rendered
110    assert "<p>Numerical and end-to-end validation suite.</p>" in rendered
111    assert r"\section*" not in rendered
112    assert r"\begin{align}" not in rendered

140def test_graph_payload_marks_nontrivial_scc_membership(tmp_path):
141    graph = GraphSpec.model_validate(
142        {
143            "project": {"id": "cycle", "title": "Cycle", "tex_root": "mathgraph/paper/main.tex"},
144            "nodes": [
145                {
146                    "id": "var.a",
147                    "kind": "variable",
148                    "title": "A",
149                    "statement": "Variable A.",
150                },
151                {
152                    "id": "assumption.b",
153                    "kind": "assumption",
154                    "title": "B",
155                    "statement": "Depends on A.",
156                    "uses": ["var.a"],
157                },
158                {
159                    "id": "assumption.c",
160                    "kind": "assumption",
161                    "title": "C",
162                    "statement": "Depends on A.",
163                    "uses": ["var.a"],
164                },
165            ],
166            "edges": [
167                {
168                    "from": "assumption.b",
169                    "to": "assumption.c",
170                    "kind": "depends_on",
171                    "description": "B depends on C.",
172                    "tex": {"file": "mathgraph/paper/derivations/init_index_products.tex", "label": "cycle:b-c"},
173                },
174                {
175                    "from": "assumption.c",
176                    "to": "assumption.b",
177                    "kind": "depends_on",
178                    "description": "C depends on B.",
179                    "tex": {"file": "mathgraph/paper/derivations/init_index_products.tex", "label": "cycle:c-b"},
180                },
181            ],
182        }
183    )
184
185    payload = _graph_payload(graph, Path(tmp_path))
186    clusters = payload["clusters"]
187    assert len(clusters) == 1
188    assert clusters[0]["size"] == 2
189    node_clusters = {node["id"]: node["cluster"] for node in payload["nodes"]}
190    assert node_clusters["var.a"] is None
191    assert node_clusters["assumption.b"]["id"] == clusters[0]["id"]
192    assert node_clusters["assumption.c"]["id"] == clusters[0]["id"]
