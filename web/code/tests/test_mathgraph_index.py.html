17def test_indexes_python_matlab_and_tex_evidence(tmp_path):
18    _write(
19        tmp_path / "model.py",
20        '''import numpy as np
21
22def estimate(x: float = 1.0):
23    """Posterior MAP estimate under a Gaussian model."""
24    assert x > 0
25    sample = np.random.normal(0, x)
26    output = "results/estimate.json"
27    return sample, output
28''',
29    )
30    _write(
31        tmp_path / "fit.m",
32        """%% Gaussian estimator
33function [mu] = fit_model(X, y)
34assert(size(X, 1) == numel(y));
35noise = randn(size(y));
36mu = (X' * X) \\ (X' * (y + noise));
37save('results/fit.mat', 'mu');
38end
39""",
40    )
41    _write(
42        tmp_path / "model.tex",
43        r"""\section{Model}
44\newcommand{\betaMAP}{\hat\beta}
45The posterior estimator is
46\begin{equation}
47\betaMAP = (X^T X)^{-1}X^T y.
48\label{eq:map}
49\end{equation}
50""",
51    )
52
53    result = build_index(tmp_path)
54    _, records = load_index(tmp_path)
55    by_path = {record["path"]: record for record in records}
56
57    assert result.passed
58    assert by_path["model.py"]["parser"] == "python-ast-tokenize-v1"
59    assert by_path["model.py"]["identifiers"]["sample"] == [6, 8]
60    assert by_path["model.py"]["assertions"] == [5]
61    assert by_path["model.py"]["defaults"][0]["name"] == "x"
62    assert by_path["model.py"]["outputs"] == [{"line": 7, "path": "results/estimate.json"}]
63    assert by_path["fit.m"]["symbols"][0]["name"] == "fit_model"
64    assert by_path["fit.m"]["assertions"] == [3]
65    assert any(region["kind"] == "mathematical-call" for region in by_path["fit.m"]["regions"])
66    assert any(item["name"] == "eq:map" for item in by_path["model.tex"]["symbols"])
67    assert any(region["kind"] == "equation" and region["start_line"] == 4 for region in by_path["model.tex"]["regions"])

151def test_indexed_regions_reduce_mixed_fixture_input_by_half(tmp_path):
152    python_boilerplate = "\n".join("# ordinary implementation detail" for _ in range(120))
153    _write(
154        tmp_path / "model.py",
155        f"{python_boilerplate}\n\ndef posterior(beta=1.0):\n    assert beta > 0\n    return beta\n",
156    )
157    matlab_boilerplate = "\n".join("% ordinary implementation detail" for _ in range(80))
158    _write(
159        tmp_path / "fit.m",
160        f"{matlab_boilerplate}\n%% MAP estimator\nfunction b = fit_map(X, y)\nb = X \\ y;\nend\n",
161    )
162    _write(
163        tmp_path / "model.tex",
164        "Prose.\n" * 50 + "\\begin{equation}\n\\hat\\beta = X^T y\n\\end{equation}\n",
165    )
166
167    result = build_index(tmp_path)
168    raw_characters = sum((tmp_path / item.path).stat().st_size for item in discover_project_files(tmp_path))
169    reviewed_characters = 0
170    for record in result.records:
171        lines = (tmp_path / record["path"]).read_text(encoding="utf-8").splitlines()
172        reviewed_lines = {
173            line
174            for region in record["regions"]
175            for line in range(region["start_line"], region["end_line"] + 1)
176        }
177        reviewed_characters += sum(len(lines[line - 1]) + 1 for line in reviewed_lines)
178
179    assert result.passed
180    assert reviewed_characters <= raw_characters * 0.5
181    assert "posterior" in result.records[1]["identifiers"] or any(
182        "posterior" in record["identifiers"] for record in result.records
183    )

217def test_generated_cache_family_is_aggregated_and_dossier_is_compact(tmp_path):
218    for index in range(10):
219        _write(tmp_path / f"src/model_{index}.py", f"def estimate_{index}(x):\n    return x + {index}\n")
220    for index in range(76):
221        _write(tmp_path / f"graphify-out/cache/item_{index}.json", '{"cached": true}\n')
222
223    result = build_index(tmp_path)
224    families = [record for record in result.records if record["record_type"] == "family"]
225    summary = format_index_summary(tmp_path)
226    dossier = format_index_dossier(tmp_path)
227
228    assert len(families) == 1
229    assert families[0]["family"] == "graphify-out/cache"
230    assert families[0]["file_count"] == 76
231    assert "item_0.json" not in summary
232    assert "graphify-out/cache: 76 files, hashes verified" in dossier
233
234    expanded = build_index(tmp_path, include_generated=True)
235    expanded_text = "\n".join(
236        format_index_file(tmp_path, record["path"])
237        for record in expanded.records
238        if record["record_type"] == "file"
239    )
240    assert len(dossier) < len(expanded_text) * 0.5

273def _write(path: Path, text: str) -> None:
274    path.parent.mkdir(parents=True, exist_ok=True)
275    path.write_text(text, encoding="utf-8")
