1"""Tests attached to Gaussian posterior, MAP, and validation graph nodes."""
2
3import numpy as np
4
5from linear_bayes.estimators import (
6    beta_map_closed_form,
7    gaussian_posterior,
8    negative_log_posterior_gaussian,
9)
10from linear_bayes.experiments import run_recovery_experiment
11from linear_bayes.simulate import simulate_gaussian_linear
12
13
14def test_closed_form_map_equals_posterior_mean():
15    """Validation attached to graph node `estimator.beta_map`."""
16    beta_true = np.array([1.0, -0.5, 0.25])
17    X, y = simulate_gaussian_linear(n=80, d=3, beta=beta_true, sigma=0.7, seed=3)
18
19    posterior_mean, posterior_cov = gaussian_posterior(X, y, sigma=0.7, tau=5.0)
20    beta_map = beta_map_closed_form(X, y, sigma=0.7, tau=5.0)
21
22    assert posterior_cov.shape == (3, 3)
23    assert np.allclose(beta_map, posterior_mean)
24
25    perturbed = beta_map + np.array([0.15, -0.1, 0.05])
26    assert negative_log_posterior_gaussian(beta_map, X, y, sigma=0.7, tau=5.0) < (
27        negative_log_posterior_gaussian(perturbed, X, y, sigma=0.7, tau=5.0)
28    )

31def test_recovery_improves_with_n():
32    """Validation attached to graph node `validation.synthetic_recovery`."""
33    beta_true = np.array([1.0, -0.5, 0.25])
34    result = run_recovery_experiment(
35        n_values=[20, 500],
36        d=3,
37        sigma=1.0,
38        tau=10.0,
39        beta_true=beta_true,
40        seeds=list(range(30)),
41    )
42
43    small_n_error, large_n_error = result["mean_errors"]
44
45    assert large_n_error < small_n_error * 0.6
46    assert set(result) == {
47        "n_values",
48        "mean_errors",
49        "std_errors",
50        "raw_errors",
51        "beta_true",
52        "sigma",
53        "tau",
54        "seeds",
55    }
