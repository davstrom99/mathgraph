15def test_student_t_simulation_has_heavy_tailed_residuals():
16    """Validation attached to graph node `model.student_t_noise`."""
17    beta_true = np.array([1.0, -0.5, 0.25])
18    X, y = simulate_student_t_linear(
19        n=4000,
20        d=3,
21        beta=beta_true,
22        sigma=1.0,
23        nu=3.0,
24        seed=11,
25    )
26
27    residual = y - X @ beta_true
28
29    assert X.shape == (4000, 3)
30    assert y.shape == (4000,)
31    assert np.mean(np.abs(residual) > 3.0) > 0.02

34def test_student_t_log_likelihood_matches_scipy_density():
35    """Validation attached to graph node `likelihood.student_t`."""
36    X = np.array([[1.0, 0.0], [1.0, 2.0], [-1.0, 1.0]])
37    y = np.array([0.5, 1.25, -0.25])
38    beta = np.array([0.4, 0.2])
39    sigma = 0.7
40    nu = 4.0
41
42    residual = y - X @ beta
43    expected = float(np.sum(t.logpdf(residual / sigma, df=nu) - np.log(sigma)))
44
45    assert np.isclose(
46        student_t_log_likelihood(X=X, y=y, beta=beta, sigma=sigma, nu=nu),
47        expected,
48    )

51def test_student_t_map_handles_outlier_better_than_gaussian_map():
52    """Validation attached to graph node `approx.student_t_map`."""
53    beta_true = np.array([1.0])
54    X = np.ones((12, 1))
55    y = np.ones(12)
56    y[-1] = 30.0
57
58    gaussian_beta = beta_map_closed_form(X=X, y=y, sigma=1.0, tau=10.0)
59    student_t_beta = student_t_map_numerical(X=X, y=y, sigma=1.0, tau=10.0, nu=3.0)
60
61    assert abs(student_t_beta[0] - beta_true[0]) < abs(gaussian_beta[0] - beta_true[0])
62    assert abs(student_t_beta[0] - beta_true[0]) < 0.25

65def test_student_t_change_script_writes_declared_outputs(tmp_path):
66    """Validation attached to graph node `experiment.student_t_change`."""
67    script_path = Path("experiments/run_student_t_change.py").resolve()
68    spec = importlib.util.spec_from_file_location("run_student_t_change", script_path)
69    assert spec is not None
70    assert spec.loader is not None
71    module = importlib.util.module_from_spec(spec)
72    spec.loader.exec_module(module)
73
74    summary_path = tmp_path / "student_t_change_summary.json"
75    plot_path = tmp_path / "student_t_change_error.png"
76    module.main(summary_path=summary_path, plot_path=plot_path)
77
78    assert summary_path.exists()
79    assert plot_path.exists()
80    assert plot_path.stat().st_size > 0
81
82    summary = json.loads(summary_path.read_text(encoding="utf-8"))
83    assert summary["nu"] == 3.0
84    assert summary["mean_student_t_error"] < summary["mean_gaussian_error"]
85    assert summary["student_t_better_fraction"] > 0.5
