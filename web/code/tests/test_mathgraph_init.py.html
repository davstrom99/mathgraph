7def test_init_empty_project_creates_valid_empty_graph(tmp_path):
8    result = initialize_mathgraph(tmp_path, project_id="empty_project", title="Empty Project")
9
10    assert result.ok
11    assert result.checklist_count == 0
12    assert (tmp_path / "mathgraph/graph.yaml").exists()
13    assert (tmp_path / "mathgraph/INIT_CHECKLIST.md").exists()
14    assert (tmp_path / "mathgraph/INIT_REPORT.md").exists()
15    assert (tmp_path / "mathgraph/INIT_INDEX.jsonl").exists()
16    assert (tmp_path / "mathgraph/paper/main.tex").exists()
17    assert not (tmp_path / "mathgraph/paper/derivations.tex").exists()
18    assert not (tmp_path / "paper").exists()
19    assert "No eligible project files were found" in (tmp_path / "mathgraph/INIT_CHECKLIST.md").read_text(encoding="utf-8")
20    report = (tmp_path / "mathgraph/INIT_REPORT.md").read_text(encoding="utf-8")
21    assert "filled by Codex during `/mathgraph init`" in report
22    assert "Initial Graph Summary" in report
23    assert "Line-level use-site coverage" in report
24    assert "Review mode: `source`" in report
25    assert (tmp_path / "mathgraph/INIT_DRAFT.yaml").exists()
26
27    check = run_checks(repo_root=tmp_path)
28    assert check.passed, format_check_result(check)
29    assert check.graph is not None
30    assert check.graph.nodes == []
31    assert check.graph.edges == []

34def test_init_existing_project_checklist_includes_eligible_files_and_excludes_generated(tmp_path):

51    assert "path` plus `line`" in checklist

53    assert "mathgraph index show <path>" in checklist
54    assert "`src/model.py`" in checklist
55    assert "`analysis.jl`" in checklist
56    assert "`paper.tex`" in checklist
57    assert "`notes.md`" in checklist
58    assert "binary artifacts" in checklist
59    assert "`web`" in checklist
60    assert "paper.aux" not in checklist
61    assert "figure.png" not in checklist
62    assert ".venv" not in checklist
63    assert "web/app.js" not in checklist

170def test_init_groups_generated_family_and_supports_explicit_include_generated(tmp_path):
171    _write(tmp_path / "src/model.py", "x = 1\n")
172    for index in range(12):
173        _write(tmp_path / f"graphify-out/cache/{index}.json", '{"x": 1}\n')
174
175    source = initialize_mathgraph(tmp_path)
176    checklist = (tmp_path / "mathgraph/INIT_CHECKLIST.md").read_text(encoding="utf-8")
177
178    assert source.review_mode == "source"
179    assert source.checklist_count == 2
180    assert "graphify-out/cache` - cache; 12 files" in checklist
181    assert "graphify-out/cache/0.json" not in checklist
182
183    included = initialize_mathgraph(tmp_path, force=True, include_generated=True)
184    included_checklist = (tmp_path / "mathgraph/INIT_CHECKLIST.md").read_text(encoding="utf-8")
185    assert included.review_mode == "include-generated"
186    assert "graphify-out/cache/0.json" in included_checklist

189def _write(path: Path, text: str) -> None:
190    path.parent.mkdir(parents=True, exist_ok=True)
191    path.write_text(text, encoding="utf-8")
