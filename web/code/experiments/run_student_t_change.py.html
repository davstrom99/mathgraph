experiments/run_student_t_change.py

1"""Run the Student-t model-change experiment.
2
3Attached graph node: `experiment.student_t_change`.
4Declared outputs:
5- results/student_t_change_summary.json
6- results/student_t_change_error.png
7"""
8
9from __future__ import annotations
10
11import json
12from pathlib import Path
13
14import matplotlib.pyplot as plt
15import numpy as np
16
17from linear_bayes.estimators import beta_map_closed_form, student_t_map_numerical
18from linear_bayes.simulate import simulate_student_t_linear
19
20
21BETA_TRUE = np.array([1.0, -0.5, 0.25])
22N = 120
23SEEDS = list(range(40))
24SIGMA = 1.0
25TAU = 10.0
26NU = 3.0
27SUMMARY_PATH = Path("results/student_t_change_summary.json")
28PLOT_PATH = Path("results/student_t_change_error.png")
29
30
31def main(
32    summary_path: Path = SUMMARY_PATH,
33    plot_path: Path = PLOT_PATH,
34) -> None:
35    """Run and save outputs for graph node `experiment.student_t_change`."""
36    gaussian_errors: list[float] = []
37    student_t_errors: list[float] = []
38
39    for seed in SEEDS:
40        X, y = simulate_student_t_linear(
41            n=N,
42            d=BETA_TRUE.size,
43            beta=BETA_TRUE,
44            sigma=SIGMA,
45            nu=NU,
46            seed=seed,
47        )
48        gaussian_beta = beta_map_closed_form(X=X, y=y, sigma=SIGMA, tau=TAU)
49        student_t_beta = student_t_map_numerical(X=X, y=y, sigma=SIGMA, tau=TAU, nu=NU)
50        gaussian_errors.append(float(np.linalg.norm(gaussian_beta - BETA_TRUE)))
51        student_t_errors.append(float(np.linalg.norm(student_t_beta - BETA_TRUE)))
52
53    summary = {
54        "beta_true": BETA_TRUE.tolist(),
55        "n": N,
56        "sigma": SIGMA,
57        "tau": TAU,
58        "nu": NU,
59        "seeds": SEEDS,
60        "gaussian_errors": gaussian_errors,
61        "student_t_errors": student_t_errors,
62        "mean_gaussian_error": float(np.mean(gaussian_errors)),
63        "mean_student_t_error": float(np.mean(student_t_errors)),
64        "median_gaussian_error": float(np.median(gaussian_errors)),
65        "median_student_t_error": float(np.median(student_t_errors)),
66        "student_t_better_fraction": float(
67            np.mean(np.asarray(student_t_errors) < np.asarray(gaussian_errors))
68        ),
69    }
70
71    summary_path.parent.mkdir(parents=True, exist_ok=True)
72    plot_path.parent.mkdir(parents=True, exist_ok=True)
73    summary_path.write_text(json.dumps(summary, indent=2), encoding="utf-8")
74
75    fig, ax = plt.subplots(figsize=(7.0, 4.5), constrained_layout=True)
76    positions = np.array([0, 1], dtype=float)
77    ax.boxplot(
78        [gaussian_errors, student_t_errors],
79        positions=positions,
80        widths=0.45,
81        patch_artist=True,
82        boxprops={"facecolor": "#dbeafe", "edgecolor": "#2563eb"},
83        medianprops={"color": "#111827", "linewidth": 2.0},
84        whiskerprops={"color": "#64748b"},
85        capprops={"color": "#64748b"},
86    )
87    for index, values in enumerate([gaussian_errors, student_t_errors]):
88        jitter = np.linspace(-0.08, 0.08, num=len(values))
89        ax.scatter(
90            np.full(len(values), positions[index]) + jitter,
91            values,
92            s=18,
93            alpha=0.65,
94            color="#0f766e" if index else "#b45309",
95        )
96    ax.set_xticks(positions, ["Gaussian MAP", "Student-t MAP"])
97    ax.set_ylabel(r"$\|\hat{\beta} - \beta_0\|_2$")
98    ax.set_title("Student-t model-change recovery under heavy-tailed noise")
99    ax.grid(True, axis="y", alpha=0.3)
100    fig.savefig(plot_path, dpi=160)
101    plt.close(fig)
102
103    print(f"Wrote {summary_path}")
104    print(f"Wrote {plot_path}")
105    print(
106        "Mean coefficient error: "
107        f"Gaussian MAP {summary['mean_gaussian_error']:.4f}, "
108        f"Student-t MAP {summary['mean_student_t_error']:.4f}"
109    )
110
111
112if __name__ == "__main__":
113    main()
