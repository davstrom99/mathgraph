6import json
7import sys

17    build_index,

19    format_draft_refs,
20    format_index_check,
21    format_index_dossier,
22    format_index_file,

31def build_parser() -> argparse.ArgumentParser:
32    parser = argparse.ArgumentParser(prog="mathgraph")
33    subparsers = parser.add_subparsers(dest="command", required=True)
34
35    init_parser = subparsers.add_parser("init", help="initialize mathgraph scaffolding")
36    init_parser.add_argument("--project-id", help="project id for the initial graph")
37    init_parser.add_argument("--title", help="project title for the initial graph")
38    init_parser.add_argument("--force", action="store_true", help="overwrite existing scaffold files")
39    init_parser.add_argument("--dry-run", action="store_true", help="show planned scaffold files without writing")
40    init_parser.add_argument("--include-extension", action="append", help="extension set to include instead of defaults; may be repeated")
41    init_parser.add_argument("--exclude-dir", action="append", help="directory name to exclude; may be repeated")
42    init_parser.add_argument(
43        "--include-generated",
44        action="store_true",
45        help="include textual generated/cache/output evidence in indexed semantic review",
46    )
47    init_parser.add_argument(
48        "--full-review",
49        action="store_true",
50        help="require complete semantic reading of every eligible file instead of indexed-region review",
51    )
52
53    index_parser = subparsers.add_parser("index", help="build and query the initialization evidence index")
54    index_subparsers = index_parser.add_subparsers(dest="index_command", required=True)
55    index_build = index_subparsers.add_parser("build", help="build or refresh mathgraph/INIT_INDEX.jsonl")
56    index_build.add_argument("--full-review", action="store_true", help="record full-review mode in the index")
57    index_build.add_argument("--include-generated", action="store_true", help="include textual generated evidence in semantic review")
58    index_summary = index_subparsers.add_parser("summary", help="show bounded index coverage and parser status")
59    index_summary.add_argument("--limit", type=int, default=100, help="maximum file records to print")
60    index_summary.add_argument("--offset", type=int, default=0, help="file record offset for pagination")
61    index_show = index_subparsers.add_parser("show", help="show significant regions for one indexed file")
62    index_show.add_argument("path", help="repository-relative indexed file path")
63    index_show.add_argument("--limit", type=int, default=40, help="maximum symbols and regions to print")
64    index_show.add_argument("--offset", type=int, default=0, help="symbol and region offset for pagination")
65    index_find = index_subparsers.add_parser("find", help="find exact indexed identifier occurrences")
66    index_find.add_argument("identifier", help="case-sensitive identifier to find")
67    index_find.add_argument("--limit", type=int, default=100, help="maximum occurrences to print")
68    index_find.add_argument("--offset", type=int, default=0, help="occurrence offset for pagination")
69    index_subparsers.add_parser("check", help="validate index coverage, parsers, and source hashes")
70    index_dossier = index_subparsers.add_parser("dossier", help="show compact cross-file mathematical evidence")
71    index_dossier.add_argument("--limit", type=int, default=200, help="maximum evidence entries to print")
72    index_dossier.add_argument("--offset", type=int, default=0, help="evidence entry offset for pagination")
73
74    refs_parser = subparsers.add_parser("draft-refs", help="emit exhaustive role-classified YAML code references")
75    refs_parser.add_argument("target", help="indexed symbol or existing graph node id")
76    refs_parser.add_argument("--output", help="write YAML to this repository-relative path instead of stdout")
77
78    check_parser = subparsers.add_parser("check", help="validate the mathgraph YAML file")
79    check_parser.add_argument(
80        "--graph",
81        default="mathgraph/graph.yaml",
82        help="path to the graph YAML file, relative to the repository root",
83    )
84
85    node_parser = subparsers.add_parser("node", help="print a graph node card")
86    node_parser.add_argument("node_id", help="node id to inspect")
87
88    impact_parser = subparsers.add_parser("impact", help="show downstream impact from a node")
89    impact_parser.add_argument("node_id", help="node id to analyze")
90    impact_parser.add_argument("--verbose", action="store_true", help="include edge paths and descriptions")
91
92    subparsers.add_parser("coverage", help="summarize graph coverage")
93    subparsers.add_parser("render", help="generate the static graph webpage in web/")
94
95    orphans_parser = subparsers.add_parser("orphans", help="report Python symbols not attached to graph nodes")
96    orphans_parser.add_argument("--root", action="append", dest="roots", help="Python root path to scan; may be repeated")
97    orphans_parser.add_argument("--exclude", action="append", dest="excludes", help="path glob or path part to exclude; may be repeated")
98    orphans_parser.add_argument("--include-private", action="store_true", help="include symbols whose names start with underscore")
99    orphans_parser.add_argument("--json", action="store_true", help="emit machine-readable JSON")
100
101    return parser
102
103
104def main(argv: list[str] | None = None) -> int:
105    parser = build_parser()
106    args = parser.parse_args(argv)
107
108    if getattr(args, "include_generated", False) and getattr(args, "full_review", False):
109        parser.error("--include-generated and --full-review are mutually exclusive")
110
111    if args.command == "init":
112        result = initialize_mathgraph(
113            find_repo_root(),
114            project_id=args.project_id,
115            title=args.title,
116            force=args.force,
117            dry_run=args.dry_run,
118            include_extensions=set(args.include_extension) if args.include_extension else None,
119            exclude_dirs=set(args.exclude_dir) if args.exclude_dir else None,
120            full_review=args.full_review,
121            include_generated=args.include_generated,
122        )
123        print(format_init_result(result))
124        return 0 if result.ok else 1
125
126    if args.command == "check":
127        result = run_checks(args.graph)
128        print(format_check_result(result))
129        return 0 if result.passed else 1
130
131    if args.command == "index":
132        root = find_repo_root()
133        try:
134            if args.index_command == "build":
135                result = build_index(root, full_review=args.full_review, include_generated=args.include_generated)
136                semantic_files = sum(1 for record in result.records if record.get("record_type") == "file")
137                families = sum(1 for record in result.records if record.get("record_type") == "family")
138                print(
139                    f"Indexed {semantic_files} semantic files and {families} aggregate families; "
140                    f"reused {result.reused}; blocked {result.blocked}; review mode {result.review_mode}."
141                )
142                return 0 if result.passed else 1
143            if args.index_command == "summary":
144                print(format_index_summary(root, limit=max(1, args.limit), offset=max(0, args.offset)))
145                return 0
146            if args.index_command == "show":
147                print(format_index_file(root, args.path, limit=max(1, args.limit), offset=max(0, args.offset)))
148                return 0
149            if args.index_command == "find":
150                print(format_index_find(root, args.identifier, limit=max(1, args.limit), offset=max(0, args.offset)))
151                return 0
152            if args.index_command == "check":
153                result = check_index(root)
154                print(format_index_check(result))
155                return 0 if result.passed else 1
156            if args.index_command == "dossier":
157                print(format_index_dossier(root, limit=max(1, args.limit), offset=max(0, args.offset)))
158                return 0
159        except ValueError as exc:
160            print(f"Error: {exc}", file=sys.stderr)
161            return 1
162
163    try:
164        if args.command == "draft-refs":
165            root = find_repo_root()
166            output = format_draft_refs(root, args.target)
167            if args.output:
168                destination = root / args.output
169                destination.parent.mkdir(parents=True, exist_ok=True)
170                destination.write_text(output, encoding="utf-8")
171                print(f"Wrote {destination.relative_to(root).as_posix()}")
172            else:
173                print(output, end="")
174            return 0
175        if args.command == "node":
176            print(format_node_card(args.node_id))
177            return 0
178        if args.command == "impact":
179            print(format_impact_report(args.node_id, verbose=args.verbose))
180            return 0
181        if args.command == "coverage":
182            print(format_coverage_report())
183            return 0
184        if args.command == "render":
185            paths = render_static_site()
186            print("Generated static graph webpage:")
187            for path in paths:
188                print(f"- {path}")
189            report_path = next((path for path in paths if path.name == "render-report.json"), None)
190            if report_path and report_path.exists():
191                report = json.loads(report_path.read_text(encoding="utf-8"))
192                titles = report.get("node_titles", [])
193                derivations = [item for item in report.get("tex", []) if item.get("kind") == "derivation"]
194                title_fallbacks = sum(item.get("status") != "rendered" for item in titles)
195                derivation_fallbacks = sum(item.get("status") == "browser-fallback" for item in derivations)
196                print("")
197                print(
198                    f"Math rendering: {len(titles) - title_fallbacks}/{len(titles)} node titles rendered locally; "
199                    f"{len(derivations) - derivation_fallbacks}/{len(derivations)} derivations compiled locally."
200                )
201                if title_fallbacks or derivation_fallbacks:
202                    print("Warning: inspect render-report.json; some math assets require fallback rendering.")
203            print("")
204            print("Serve it with:")
205            print("  python -m http.server 8000 --bind 127.0.0.1 -d web")
206            print("")
207            print("Then open:")
208            print("  http://127.0.0.1:8000/")
209            print("")
210            print("The server command stays running; press Ctrl+C in that terminal to stop it.")
211            return 0
212        if args.command == "orphans":
213            report = find_orphans(
214                roots=args.roots,
215                excludes=args.excludes,
216                include_private=args.include_private,
217            )
218            print(format_orphan_report(report, as_json=args.json))
219            return 0 if report.passed else 1
220    except ValueError as exc:
221        print(f"Error: {exc}", file=sys.stderr)
222        return 1
223
224    parser.error(f"unknown command: {args.command}")
225    return 2
226
227
228if __name__ == "__main__":
229    sys.exit(main())
