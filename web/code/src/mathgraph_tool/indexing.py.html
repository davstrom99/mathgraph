50@dataclass(frozen=True)
51class IndexBuildResult:
52    records: list[dict[str, Any]]
53    reused: int
54    blocked: int
55    review_mode: str
56    include_extensions: tuple[str, ...] = ()
57    exclude_dirs: tuple[str, ...] = ()
58    classification: dict[str, tuple[str, ...]] | None = None
59
60    @property
61    def passed(self) -> bool:
62        return self.review_mode == "full" or self.blocked == 0

76def build_index(
77    repo_root: Path,
78    *,
79    files: list[DiscoveredFile] | None = None,
80    full_review: bool = False,
81    write: bool = True,
82    index_path: Path = INDEX_PATH,
83    include_extensions: set[str] | None = None,
84    exclude_dirs: set[str] | None = None,
85    include_generated: bool = False,
86) -> IndexBuildResult:
87    """Build or refresh the initialization evidence index."""
88    root = repo_root.resolve()
89    discovered = files if files is not None else discover_project_files(
90        root, include_extensions=include_extensions, exclude_dirs=exclude_dirs
91    )
92    destination = root / index_path
93    previous = _load_file_records(destination)
94    records: list[dict[str, Any]] = []
95    reused = 0
96
97    family_members: dict[tuple[str, str], list[dict[str, Any]]] = {}
98    mode = "full" if full_review else "include-generated" if include_generated else "source"
99    for item in discovered:
100        source = root / item.path
101        digest = _sha256(source)
102        semantic = item.evidence_class == "authoritative" or (
103            include_generated and item.evidence_class in {"generated", "cache", "rendered", "output"} and item.category != "binary"
104        ) or full_review
105        if not semantic:
106            key = (item.evidence_class, item.family or item.path.parent.as_posix() or ".")
107            family_members.setdefault(key, []).append(
108                {"path": item.path.as_posix(), "sha256": digest, "size_bytes": source.stat().st_size}
109            )
110            continue
111        cached = previous.get(item.path.as_posix())
112        if (
113            cached
114            and cached.get("sha256") == digest
115            and cached.get("schema_version") == INDEX_SCHEMA_VERSION
116            and cached.get("parser_version") == PARSER_VERSION
117        ):
118            record = cached
119            reused += 1
120        else:
121            record = _index_file(source, item, digest)
122        records.append(record)
123
124    for (evidence_class, family), members in sorted(family_members.items()):
125        members.sort(key=lambda member: member["path"])
126        records.append(_family_record(evidence_class, family, members))
127
128    records.sort(key=lambda record: (record.get("record_type") != "file", record.get("path", record.get("family", ""))))
129    config = load_discovery_config(root)
130    result = IndexBuildResult(
131        records=records,
132        reused=reused,
133        blocked=sum(record["status"] == "blocked" for record in records),
134        review_mode=mode,
135        include_extensions=tuple(sorted(include_extensions or set())),
136        exclude_dirs=tuple(sorted(exclude_dirs or set())),
137        classification={
138            "include": config.include,
139            "exclude": config.exclude,
140            "authoritative": config.authoritative,
141            "generated": config.generated,
142        },
143    )
144    if write:
145        destination.parent.mkdir(parents=True, exist_ok=True)
146        destination.write_text(serialize_index(result), encoding="utf-8")
147    return result

150def serialize_index(result: IndexBuildResult) -> str:
151    meta = {
152        "record_type": "meta",
153        "schema_version": INDEX_SCHEMA_VERSION,
154        "parser_version": PARSER_VERSION,
155        "review_mode": result.review_mode,
156        "file_count": len(result.records),
157        "include_extensions": list(result.include_extensions),
158        "exclude_dirs": list(result.exclude_dirs),
159        "classification": {key: list(value) for key, value in (result.classification or {}).items()},
160    }
161    lines = [json.dumps(meta, sort_keys=True)]
162    lines.extend(json.dumps(record, sort_keys=True) for record in result.records)
163    return "\n".join(lines) + "\n"

166def load_index(repo_root: Path, index_path: Path = INDEX_PATH) -> tuple[dict[str, Any], list[dict[str, Any]]]:
167    path = repo_root.resolve() / index_path
168    if not path.exists():
169        raise ValueError(f"initialization index does not exist: {index_path.as_posix()}")
170    lines = [line for line in path.read_text(encoding="utf-8").splitlines() if line.strip()]
171    if not lines:
172        raise ValueError("initialization index is empty")
173    try:
174        items = [json.loads(line) for line in lines]
175    except json.JSONDecodeError as exc:
176        raise ValueError(f"invalid initialization index JSON: {exc}") from exc
177    if items[0].get("record_type") != "meta":
178        raise ValueError("initialization index is missing its metadata record")
179    return items[0], items[1:]

182def check_index(repo_root: Path, index_path: Path = INDEX_PATH) -> IndexCheckResult:
183    root = repo_root.resolve()
184    ok: list[str] = []
185    warnings: list[str] = []
186    errors: list[str] = []
187    try:
188        meta, records = load_index(root, index_path)
189    except ValueError as exc:
190        return IndexCheckResult(ok, warnings, [str(exc)])
191
192    if meta.get("schema_version") != INDEX_SCHEMA_VERSION:
193        errors.append(f"unsupported index schema version: {meta.get('schema_version')}")
194    if meta.get("parser_version") != PARSER_VERSION:
195        errors.append(f"stale index parser version: {meta.get('parser_version')}")
196    mode = meta.get("review_mode")
197    if mode not in {"source", "include-generated", "full"}:
198        errors.append(f"invalid index review mode: {mode}")
199
200    includes = set(meta.get("include_extensions") or []) or None
201    excludes = set(meta.get("exclude_dirs") or []) or None
202    current = build_index(
203        root,
204        write=False,
205        include_extensions=includes,
206        exclude_dirs=excludes,
207        include_generated=mode == "include-generated",
208        full_review=mode == "full",
209    )
210    expected_by_key = {_record_key(record): record for record in current.records}
211    indexed_by_key = {_record_key(record): record for record in records}
212    for key in sorted(expected_by_key.keys() - indexed_by_key.keys()):
213        errors.append(f"eligible evidence missing from index: {_display_record_key(key)}")
214    for key in sorted(indexed_by_key.keys() - expected_by_key.keys()):
215        errors.append(f"index contains ineligible or removed evidence: {_display_record_key(key)}")
216
217    blocked: list[str] = []
218    for record in records:
219        if record.get("record_type") == "family":
220            expected = expected_by_key.get(_record_key(record))
221            if expected and (
222                record.get("sha256") != expected.get("sha256")
223                or record.get("file_count") != expected.get("file_count")
224            ):
225                errors.append(f"indexed evidence family is stale: {record.get('family')}")
226            continue
227        relative = record.get("path", "")
228        source = root / relative
229        if not source.exists():
230            errors.append(f"indexed file missing: {relative}")
231            continue
232        if record.get("sha256") != _sha256(source):
233            errors.append(f"indexed file is stale: {relative}")
234        if record.get("status") == "blocked":
235            blocked.append(relative)
236
237    if blocked and mode != "full":
238        errors.extend(f"indexed review blocked by unsupported or malformed file: {path}" for path in blocked)
239    elif blocked:
240        warnings.extend(f"file requires complete semantic review: {path}" for path in blocked)
241
242    if not errors:
243        file_count = sum(1 for record in records if record.get("record_type") == "file")
244        family_count = sum(1 for record in records if record.get("record_type") == "family")
245        ok.append(f"index covers {file_count} semantic files and {family_count} aggregate families with current hashes")
246    if not blocked:
247        ok.append("all indexed files have supported deterministic parsers")
248    return IndexCheckResult(ok, warnings, errors)

251def _record_key(record: dict[str, Any]) -> tuple[str, str]:
252    if record.get("record_type") == "family":
253        return "family", f"{record.get('evidence_class')}:{record.get('family')}"
254    return "file", str(record.get("path"))

261def format_index_summary(repo_root: Path, limit: int = 100, offset: int = 0) -> str:
262    meta, records = load_index(repo_root)
263    status_counts: dict[str, int] = {}
264    for record in records:
265        status_counts[record["status"]] = status_counts.get(record["status"], 0) + 1
266    lines = [
267        f"Review mode: {meta['review_mode']}",
268        f"Files: {len(records)}",
269        "Status: " + ", ".join(f"{key}={value}" for key, value in sorted(status_counts.items())),
270        "",
271    ]
272    page = records[offset:offset + limit]
273    for record in page:
274        if record.get("record_type") == "family":
275            lines.append(
276                f"{record['family']} [{record['evidence_class']}/aggregate] "
277                f"files={record['file_count']} bytes={record['size_bytes']} hashes=verified"
278            )
279        else:
280            lines.append(
281                f"{record['path']} [{record['language']}/{record['status']}] "
282                f"lines={record['line_count']} regions={len(record['regions'])} "
283                f"symbols={len(record['symbols'])}"
284            )
285    remaining = max(0, len(records) - offset - len(page))
286    if remaining:
287        lines.append(f"... {remaining} more files; rerun with `--offset {offset + len(page)}`.")
288    return "\n".join(lines)

291def format_index_file(repo_root: Path, relative_path: str, limit: int = 40, offset: int = 0) -> str:
292    root = repo_root.resolve()
293    _, records = load_index(root)
294    normalized = Path(relative_path).as_posix()
295    try:
296        record = next(item for item in records if item["path"] == normalized)
297    except StopIteration as exc:
298        raise ValueError(f"file is not present in initialization index: {normalized}") from exc
299    source_lines = _safe_source_lines(root / normalized)
300    lines = [
301        f"File: {normalized}",
302        f"Parser: {record['parser']} ({record['status']})",
303        f"Hash: {record['sha256']}",
304    ]
305    if record.get("error"):
306        lines.append(f"Error: {record['error']}")
307    lines.append("Symbols:")
308    lines.extend(
309        f"- {symbol['kind']} {symbol['name']} lines {symbol['start_line']}-{symbol['end_line']}"
310        for symbol in record["symbols"][offset:offset + limit]
311    )
312    lines.append("Significant regions:")
313    page = record["regions"][offset:offset + limit]
314    for region in page:
315        snippet = _line_excerpt(source_lines, region["start_line"], region["end_line"])
316        lines.append(
317            f"- {region['kind']} lines {region['start_line']}-{region['end_line']}: {snippet}"
318        )
319    omitted = max(0, len(record["regions"]) - offset - len(page))
320    if omitted:
321        lines.append(f"... {omitted} more regions; rerun with `--offset {offset + len(page)}`.")
322    return "\n".join(lines)

325def format_index_find(repo_root: Path, identifier: str, limit: int = 100, offset: int = 0) -> str:
326    root = repo_root.resolve()
327    _, records = load_index(root)
328    matches: list[tuple[str, int]] = []
329    for record in records:
330        for line in record.get("identifiers", {}).get(identifier, []):
331            matches.append((record["path"], line))
332    matches.sort()
333    lines = [f"Identifier: {identifier}", f"Occurrences: {len(matches)}"]
334    source_cache: dict[str, list[str]] = {}
335    page = matches[offset:offset + limit]
336    for path, line in page:
337        source_lines = source_cache.setdefault(path, _safe_source_lines(root / path))
338        lines.append(f"- {path}:{line}: {_line_excerpt(source_lines, line, line)}")
339    remaining = max(0, len(matches) - offset - len(page))
340    if remaining:
341        lines.append(f"... {remaining} more occurrences; rerun with `--offset {offset + len(page)}`.")
342    return "\n".join(lines)

345def format_index_dossier(repo_root: Path, limit: int = 200, offset: int = 0) -> str:
346    """Emit compact cross-file mathematical evidence without repeated file headers."""
347    root = repo_root.resolve()
348    meta, records = load_index(root)
349    entries: list[tuple[str, str, int, str]] = []
350    identity_paths: dict[str, set[str]] = {}
351    defined_names = {
352        symbol["name"]
353        for record in records
354        if record.get("record_type") == "file"
355        for symbol in record.get("symbols", [])
356    }
357    for record in records:
358        if record.get("record_type") != "file":
359            continue
360        path = record["path"]
361        source = _safe_source_lines(root / path)
362        for symbol in record.get("symbols", []):
363            section = "Tests" if symbol.get("name", "").startswith("test") else "Definitions and signatures"
364            line = int(symbol["start_line"])
365            entries.append((section, path, line, _line_excerpt(source, line, line)))
366            identity_paths.setdefault(symbol["name"], set()).add(path)
367        for region in record.get("regions", []):
368            if region.get("kind") == "equation":
369                line = int(region["start_line"])
370                entries.append(("Equations", path, line, _line_excerpt(source, line, int(region["end_line"]))))
371        for default in record.get("defaults", []):
372            entries.append(("Configuration defaults", path, int(default["line"]), f"{default['name']}={default.get('value')}"))
373        for call in record.get("calls", []):
374            if call["name"] in defined_names:
375                entries.append(("Call relationships", path, int(call["line"]), f"{call.get('caller') or '<module>'} -> {call['name']}"))
376    for name, paths in identity_paths.items():
377        if len(paths) > 1:
378            entries.append(("Candidate identities", ", ".join(sorted(paths)), 0, name))
379    section_order = {
380        "Equations": 0,
381        "Definitions and signatures": 1,
382        "Call relationships": 2,
383        "Tests": 3,
384        "Configuration defaults": 4,
385        "Candidate identities": 5,
386    }
387    unique = sorted(set(entries), key=lambda item: (section_order[item[0]], item[1], item[2], item[3]))
388    page = unique[offset:offset + limit]
389    lines = [f"Review mode: {meta['review_mode']}", f"Evidence entries: {len(unique)}"]
390    current = None
391    for section, path, line, detail in page:
392        if section != current:
393            lines.extend(["", f"## {section}"])
394            current = section
395        location = f"{path}:{line}" if line else path
396        lines.append(f"- {location}: {detail}")
397    remaining = max(0, len(unique) - offset - len(page))
398    if remaining:
399        lines.append(f"\n... {remaining} more entries; rerun with `--offset {offset + len(page)}`.")
400    families = [record for record in records if record.get("record_type") == "family"]
401    if families:
402        lines.extend(["", "## Aggregate-verified families"])
403        lines.extend(
404            f"- {record['family']}: {record['file_count']} files, hashes verified ({record['evidence_class']})"
405            for record in families
406        )
407    return "\n".join(lines)

410def draft_code_refs(repo_root: Path, target: str) -> dict[str, Any]:
411    """Create exhaustive, role-classified exact-line reference YAML data."""
412    root = repo_root.resolve()
413    identifier = target
414    if "." in target and (root / "mathgraph/graph.yaml").exists():
415        from mathgraph_tool.loader import load_graph
416
417        graph = load_graph(repo_root=root)
418        node = graph.node_by_id.get(target)
419        if node is not None:
420            identifier = node.symbol or next((ref.symbol for ref in node.code if ref.symbol), "") or target.split(".")[-1]
421    _, records = load_index(root)
422    refs: dict[tuple[str, int], dict[str, Any]] = {}
423    priority = {"definition": 5, "test": 4, "experiment": 3, "configuration": 2, "caller": 1, "implementation": 0}
424    for record in records:
425        if record.get("record_type") != "file" or record.get("evidence_class") != "authoritative":
426            continue
427        path = record["path"]
428        occurrences = [item for item in record.get("occurrences", []) if item.get("name") == identifier]
429        if not occurrences:
430            occurrences = [
431                {"name": identifier, "line": line, "role": "implementation", "enclosing": None}
432                for line in record.get("identifiers", {}).get(identifier, [])
433            ]
434        for occurrence in occurrences:
435            role = _reference_role(path, occurrence.get("role", "implementation"))
436            value = {"path": path, "line": int(occurrence["line"]), "role": role}
437            if role == "definition":
438                value["symbol"] = identifier
439            key = (path, value["line"])
440            if key not in refs or priority[role] > priority[refs[key]["role"]]:
441                refs[key] = value
442    return {"target": target, "identifier": identifier, "code": [refs[key] for key in sorted(refs)]}

445def format_draft_refs(repo_root: Path, target: str) -> str:
446    return yaml.safe_dump(draft_code_refs(repo_root, target), sort_keys=False, allow_unicode=True)

449def serialize_initial_draft(result: IndexBuildResult, repo_root: Path | None = None) -> str:
450    """Generate deterministic graph candidates for later semantic resolution."""
451    symbols: list[dict[str, Any]] = []
452    tex_labels: list[dict[str, Any]] = []
453    relationships: list[dict[str, Any]] = []
454    equations: list[dict[str, Any]] = []
455    identities: dict[str, list[str]] = {}
456    for record in result.records:
457        if record.get("record_type") != "file":
458            continue
459        path = record["path"]
460        for symbol in record.get("symbols", []):
461            candidate = {"name": symbol["name"], "kind": symbol["kind"], "path": path, "line": symbol["start_line"]}
462            symbols.append(candidate)
463            identities.setdefault(symbol["name"], []).append(f"{path}:{symbol['start_line']}")
464            if record.get("language") == "tex" and symbol.get("kind") == "label":
465                tex_labels.append(candidate)
466        for call in record.get("calls", []):
467            relationships.append({"caller": call.get("caller"), "callee": call["name"], "path": path, "line": call["line"]})
468        equations.extend(
469            {"path": path, "start_line": region["start_line"], "end_line": region["end_line"]}
470            for region in record.get("regions", [])
471            if region.get("kind") == "equation"
472        )
473    provenance = _existing_provenance_candidates(repo_root, result.records)
474    payload = {
475        "review_mode": result.review_mode,
476        "generated_by": "mathgraph init",
477        "candidates": {
478            "symbols": symbols,
479            "tex_labels": tex_labels,
480            "equations": equations,
481            "calls": relationships,
482            "provenance": provenance,
483        },
484        "unresolved_identities": [
485            {"name": name, "occurrences": locations}
486            for name, locations in sorted(identities.items())
487            if len(locations) > 1
488        ],
489    }
490    return yaml.safe_dump(payload, sort_keys=False, allow_unicode=True)

493def _existing_provenance_candidates(repo_root: Path | None, records: list[dict[str, Any]]) -> list[dict[str, Any]]:
494    candidates: list[dict[str, Any]] = []
495    if repo_root is None:
496        return candidates
497    root = repo_root.resolve()
498    paths = [root / "mathgraph/graph.yaml", root / "graphify-out/graph.json"]
499    indexed_hashes = {
500        member["path"]: member["sha256"]
501        for record in records
502        if record.get("record_type") == "family"
503        for member in record.get("members", [])
504    }
505    for path in paths:
506        if not path.exists():
507            continue
508        try:
509            data = yaml.safe_load(path.read_text(encoding="utf-8"))
510        except (OSError, UnicodeDecodeError, yaml.YAMLError):
511            continue
512        relative = path.relative_to(root).as_posix()
513        candidates.append(
514            {
515                "path": relative,
516                "sha256": indexed_hashes.get(relative) or _sha256(path),
517                "nodes": len(data.get("nodes", [])) if isinstance(data, dict) else 0,
518                "edges": len(data.get("edges", [])) if isinstance(data, dict) else 0,
519            }
520        )
521    return candidates

524def _reference_role(path: str, indexed_role: str) -> str:
525    normalized = path.replace("\\", "/")
526    if normalized.startswith("tests/") or "/tests/" in f"/{normalized}":
527        return "test"
528    if normalized.startswith("experiments/") or "/experiments/" in f"/{normalized}":
529        return "experiment"
530    if Path(normalized).suffix.lower() in {".toml", ".yaml", ".yml", ".json"}:
531        return "configuration"
532    return indexed_role if indexed_role in {"definition", "caller"} else "implementation"

535def format_index_check(result: IndexCheckResult) -> str:
536    lines = [f"OK: {item}." for item in result.ok]
537    if result.warnings:
538        lines.extend(["", "Warnings:", *(f"- {item}" for item in result.warnings)])
539    if result.errors:
540        lines.extend(["", "Errors:", *(f"- {item}" for item in result.errors)])
541    return "\n".join(lines)

544def indexed_line_exists(records: Iterable[dict[str, Any]], path: str, line: int) -> bool:
545    normalized = Path(path).as_posix()
546    for record in records:
547        if record.get("path") == normalized:
548            return 1 <= line <= int(record.get("line_count", 0))
549    return False

552def _index_file(source: Path, item: DiscoveredFile, digest: str) -> dict[str, Any]:
553    suffix = source.suffix.lower()
554    if suffix == ".py":
555        parser, language, extractor = "python-ast-tokenize-v1", "python", _extract_python
556    elif suffix == ".m":
557        parser, language, extractor = "matlab-lexer-v1", "matlab", _extract_matlab
558    elif suffix == ".tex":
559        parser, language, extractor = "tex-scanner-v1", "tex", _extract_tex
560    elif suffix == ".json":
561        parser, language, extractor = "json-stdlib-v1", "json", _extract_json
562    elif suffix == ".toml":
563        parser, language, extractor = "tomllib-v1", "toml", _extract_toml
564    elif suffix in TEXT_DOCUMENT_EXTENSIONS:
565        parser, language, extractor = "generic-text-v1", suffix.lstrip("."), _extract_generic
566    elif suffix in BINARY_DOCUMENT_EXTENSIONS or suffix in CODE_EXTENSIONS:
567        return _blocked_record(source, item, digest, suffix.lstrip(".") or "unknown", "no supported deterministic parser")
568    else:
569        return _blocked_record(source, item, digest, suffix.lstrip(".") or "unknown", "unsupported file type")
570
571    try:
572        text = source.read_text(encoding="utf-8-sig")
573    except (UnicodeDecodeError, OSError) as exc:
574        return _blocked_record(source, item, digest, language, f"could not read UTF-8 text: {exc}")
575    try:
576        payload = extractor(text)
577    except (SyntaxError, tokenize.TokenError, ValueError) as exc:
578        return _blocked_record(source, item, digest, language, str(exc), line_count=len(text.splitlines()))
579    return {
580        "record_type": "file",
581        "schema_version": INDEX_SCHEMA_VERSION,
582        "parser_version": PARSER_VERSION,
583        "path": item.path.as_posix(),
584        "category": item.category,
585        "evidence_class": item.evidence_class,
586        "language": language,
587        "parser": parser,
588        "status": "ok",
589        "error": None,
590        "sha256": digest,
591        "size_bytes": source.stat().st_size,
592        "line_count": len(text.splitlines()),
593        **payload,
594    }

632def _family_record(evidence_class: str, family: str, members: list[dict[str, Any]]) -> dict[str, Any]:
633    digest = hashlib.sha256()
634    for member in members:

651def _extract_python(text: str) -> dict[str, Any]:
652    tree = ast.parse(text)
653    lines = text.splitlines()
654    symbols: list[dict[str, Any]] = []
655    regions: list[dict[str, Any]] = []
656    tests: list[dict[str, Any]] = []
657    assertions: list[int] = []
658    defaults: list[dict[str, Any]] = []
659    outputs: list[dict[str, Any]] = []
660
661    for node in ast.walk(tree):
662        if isinstance(node, ast.FunctionDef | ast.AsyncFunctionDef | ast.ClassDef):
663            end = getattr(node, "end_lineno", node.lineno)
664            kind = "class" if isinstance(node, ast.ClassDef) else "function"
665            symbols.append(_symbol(node.name, kind, node.lineno, end))
666            regions.append(_region("signature", node.lineno, node.lineno))
667            if node.name.startswith("test"):
668                tests.append(_symbol(node.name, kind, node.lineno, end))
669            body = getattr(node, "body", [])
670            if body and isinstance(body[0], ast.Expr) and isinstance(body[0].value, ast.Constant) and isinstance(body[0].value.value, str):
671                regions.append(_region("docstring", body[0].lineno, getattr(body[0], "end_lineno", body[0].lineno)))
672            if isinstance(node, ast.FunctionDef | ast.AsyncFunctionDef):
673                positional = list(node.args.posonlyargs) + list(node.args.args)
674                for argument, default in zip(positional[-len(node.args.defaults):], node.args.defaults):
675                    defaults.append({"name": argument.arg, "line": default.lineno, "value": ast.get_source_segment(text, default)})
676                    regions.append(_region("default", default.lineno, getattr(default, "end_lineno", default.lineno)))
677                for argument, default in zip(node.args.kwonlyargs, node.args.kw_defaults):
678                    if default is not None:
679                        defaults.append({"name": argument.arg, "line": default.lineno, "value": ast.get_source_segment(text, default)})
680                        regions.append(_region("default", default.lineno, getattr(default, "end_lineno", default.lineno)))
681        elif isinstance(node, ast.Assert):
682            assertions.append(node.lineno)
683            regions.append(_region("assertion", node.lineno, getattr(node, "end_lineno", node.lineno)))
684        elif isinstance(node, ast.Assign | ast.AnnAssign | ast.AugAssign):
685            regions.append(_region("assignment", node.lineno, getattr(node, "end_lineno", node.lineno)))
686        elif isinstance(node, ast.If | ast.While):
687            regions.append(_region("condition", node.lineno, getattr(node.test, "end_lineno", node.lineno)))
688        elif isinstance(node, ast.Call):
689            name = _python_call_name(node.func)
690            if MATH_KEYWORDS.search(name) or re.search(r"(minimize|normal|random|sample|solve|fit|predict|logpdf)", name, re.I):
691                regions.append(_region("mathematical-call", node.lineno, getattr(node, "end_lineno", node.lineno)))
692        elif isinstance(node, ast.Constant) and isinstance(node.value, str):
693            if Path(node.value).suffix.lower() in OUTPUT_SUFFIXES:
694                outputs.append({"path": node.value, "line": node.lineno})
695
696    comments: list[tuple[int, str]] = []
697    identifiers: dict[str, set[int]] = {}
698    for token in tokenize.generate_tokens(io.StringIO(text).readline):
699        if token.type == tokenize.NAME:
700            identifiers.setdefault(token.string, set()).add(token.start[0])
701        elif token.type == tokenize.COMMENT:
702            comments.append((token.start[0], token.string))
703    for line, comment in comments:
704        if MATH_KEYWORDS.search(comment) or any(marker in comment for marker in ("=", "~", "->", "<=")):
705            regions.append(_region("mathematical-comment", line, line))
706    occurrences, calls = _python_occurrences(tree)
707    return _payload(symbols, identifiers, regions, tests, assertions, defaults, outputs, occurrences, calls)

710def _python_occurrences(tree: ast.AST) -> tuple[list[dict[str, Any]], list[dict[str, Any]]]:
711    occurrences: list[dict[str, Any]] = []
712    calls: list[dict[str, Any]] = []
713
714    class Visitor(ast.NodeVisitor):
715        def __init__(self) -> None:
716            self.scope: list[str] = []
717
718        def visit_FunctionDef(self, node: ast.FunctionDef) -> None:
719            occurrences.append({"name": node.name, "line": node.lineno, "role": "definition", "enclosing": self.enclosing})
720            self.scope.append(node.name)
721            self.generic_visit(node)
722            self.scope.pop()
723
724        visit_AsyncFunctionDef = visit_FunctionDef
725
726        def visit_ClassDef(self, node: ast.ClassDef) -> None:
727            occurrences.append({"name": node.name, "line": node.lineno, "role": "definition", "enclosing": self.enclosing})
728            self.scope.append(node.name)
729            self.generic_visit(node)
730            self.scope.pop()
731
732        def visit_Name(self, node: ast.Name) -> None:
733            role = "definition" if isinstance(node.ctx, (ast.Store, ast.Param)) else "implementation"
734            occurrences.append({"name": node.id, "line": node.lineno, "role": role, "enclosing": self.enclosing})
735
736        def visit_Call(self, node: ast.Call) -> None:
737            name = _python_call_name(node.func).split(".")[-1]
738            if name:
739                calls.append({"name": name, "line": node.lineno, "caller": self.enclosing})
740                occurrences.append({"name": name, "line": node.lineno, "role": "caller", "enclosing": self.enclosing})
741            self.generic_visit(node)
742
743        @property
744        def enclosing(self) -> str | None:
745            return ".".join(self.scope) if self.scope else None
746
747    Visitor().visit(tree)
748    unique = {(item["name"], item["line"], item["role"], item["enclosing"]): item for item in occurrences}
749    return sorted(unique.values(), key=lambda item: (item["line"], item["name"], item["role"])), calls

752def _extract_matlab(text: str) -> dict[str, Any]:
753    lines = text.splitlines()
754    if text.count("%{") != text.count("%}"):
755        raise ValueError("unbalanced MATLAB block comment")
756    symbols: list[dict[str, Any]] = []
757    regions: list[dict[str, Any]] = []
758    tests: list[dict[str, Any]] = []
759    assertions: list[int] = []
760    defaults: list[dict[str, Any]] = []
761    outputs: list[dict[str, Any]] = []
762    identifiers: dict[str, set[int]] = {}
763    function_re = re.compile(
764        r"^\s*function\s+(?:\[[^]]*\]|\w+)\s*=\s*(\w+)"
765        r"|^\s*function\s+(\w+)(?:\s*\(|\s*$)"
766    )
767    class_re = re.compile(r"^\s*classdef(?:\s*\([^)]*\))?\s+(\w+)")
768    assignment_re = re.compile(r"^\s*([A-Za-z_]\w*)\s*=\s*(.+?);?\s*$")
769    in_block_comment = False
770    default_window = 0
771
772    for number, raw in enumerate(lines, start=1):
773        stripped = raw.strip()
774        if stripped.startswith("%{"):
775            in_block_comment = True
776        if not in_block_comment:
777            code = _matlab_code_part(raw)
778            for identifier in IDENTIFIER_RE.findall(code):
779                identifiers.setdefault(identifier, set()).add(number)
780            match = function_re.search(code)
781            if match:
782                name = match.group(1) or match.group(2)
783                symbol = _symbol(name, "function", number, number)
784                symbols.append(symbol)
785                regions.append(_region("signature", number, number))
786                if name.lower().startswith("test"):
787                    tests.append(symbol)
788            class_match = class_re.search(code)
789            if class_match:
790                symbols.append(_symbol(class_match.group(1), "class", number, number))
791                regions.append(_region("signature", number, number))
792            assignment = assignment_re.search(code)
793            if assignment:
794                regions.append(_region("assignment", number, number))
795                if default_window > 0:
796                    defaults.append({"name": assignment.group(1), "line": number, "value": assignment.group(2).rstrip(";")})
797            if re.search(r"\b(if|elseif|while)\b", code):
798                regions.append(_region("condition", number, number))
799            if re.search(r"\b(nargin|isempty|exist\s*\()", code):
800                default_window = 5
801            if re.search(r"\bassert\s*\(", code):
802                assertions.append(number)
803                regions.append(_region("assertion", number, number))
804            if re.search(r"\b(fmin|fit|log|mean|norm|optim|rand|sample|solve|std|var)\w*\s*\(", code, re.I):
805                regions.append(_region("mathematical-call", number, number))
806            for value in re.findall(r"['\"]([^'\"]+)['\"]", code):
807                if Path(value).suffix.lower() in OUTPUT_SUFFIXES:
808                    outputs.append({"path": value, "line": number})
809            default_window = max(0, default_window - 1)
810        if stripped.startswith("%%") or (stripped.startswith("%") and (MATH_KEYWORDS.search(stripped) or "=" in stripped)):
811            regions.append(_region("mathematical-comment", number, number))
812        if stripped.endswith("%}"):
813            in_block_comment = False
814    return _payload(symbols, identifiers, regions, tests, assertions, defaults, outputs)

817def _extract_tex(text: str) -> dict[str, Any]:
818    lines = text.splitlines()
819    begin_re = re.compile(r"\\begin\{([^}]+)\}")
820    end_re = re.compile(r"\\end\{([^}]+)\}")
821    math_envs = {"align", "align*", "equation", "equation*", "gather", "gather*", "multline", "multline*"}
822    stack: list[tuple[str, int]] = []
823    symbols: list[dict[str, Any]] = []
824    regions: list[dict[str, Any]] = []
825    identifiers: dict[str, set[int]] = {}
826
827    for number, line in enumerate(lines, start=1):
828        for identifier in re.findall(r"\\([A-Za-z]+)|\b([A-Za-z]\w*)\b", line):
829            value = identifier[0] or identifier[1]
830            identifiers.setdefault(value, set()).add(number)
831        for label in re.findall(r"\\label\{([^}]+)\}", line):
832            symbols.append(_symbol(label, "label", number, number))
833        if re.search(r"\\(newcommand|renewcommand|DeclareMathOperator)\b", line):
834            regions.append(_region("definition", number, number))
835        if re.search(r"\\(part|chapter|section|subsection|paragraph)\{", line):
836            regions.append(_region("heading", number, number))
837        if re.search(r"\\(input|include)\{", line):
838            regions.append(_region("include", number, number))
839        if re.search(r"\\(ref|eqref|cite)\{", line):
840            regions.append(_region("reference", number, number))
841        if MATH_KEYWORDS.search(line):
842            regions.append(_region("mathematical-prose", number, number))
843        for env in begin_re.findall(line):
844            stack.append((env, number))
845        for env in end_re.findall(line):
846            if not stack or stack[-1][0] != env:
847                raise ValueError(f"unbalanced TeX environment {env} at line {number}")
848            opened, start = stack.pop()
849            if opened in math_envs:
850                regions.append(_region("equation", start, number))
851        if "\\[" in line or "\\(" in line or "$" in line:
852            regions.append(_region("inline-math", number, number))
853    if stack:
854        env, line = stack[-1]
855        raise ValueError(f"unclosed TeX environment {env} from line {line}")
856    return _payload(symbols, identifiers, regions, [], [], [], [])

859def _extract_generic(text: str) -> dict[str, Any]:
860    symbols: list[dict[str, Any]] = []
861    regions: list[dict[str, Any]] = []
862    identifiers: dict[str, set[int]] = {}
863    outputs: list[dict[str, Any]] = []
864    for number, line in enumerate(text.splitlines(), start=1):
865        for identifier in IDENTIFIER_RE.findall(line):
866            identifiers.setdefault(identifier, set()).add(number)
867        stripped = line.strip()
868        if re.match(r"^#{1,6}\s+", stripped) or re.match(r"^\[[^]]+\]\s*$", stripped):
869            regions.append(_region("heading", number, number))
870        if MATH_KEYWORDS.search(line) or any(marker in line for marker in ("\\[", "$$", " := ", " ~ ")):
871            regions.append(_region("mathematical-text", number, number))
872        for value in re.findall(r"['\"]([^'\"]+)['\"]", line):
873            if Path(value).suffix.lower() in OUTPUT_SUFFIXES:
874                outputs.append({"path": value, "line": number})
875    return _payload(symbols, identifiers, regions, [], [], [], outputs)

888def _payload(symbols, identifiers, regions, tests, assertions, defaults, outputs, occurrences=None, calls=None) -> dict[str, Any]:
889    deduped_regions = {
890        (region["kind"], region["start_line"], region["end_line"]): region
891        for region in regions
892    }
893    return {
894        "symbols": sorted(symbols, key=lambda item: (item["start_line"], item["name"])),
895        "identifiers": {name: sorted(lines) for name, lines in sorted(identifiers.items())},
896        "regions": sorted(deduped_regions.values(), key=lambda item: (item["start_line"], item["end_line"], item["kind"])),
897        "tests": sorted(tests, key=lambda item: (item["start_line"], item["name"])),
898        "assertions": sorted(set(assertions)),
899        "defaults": sorted(defaults, key=lambda item: (item["line"], item["name"])),
900        "outputs": sorted(outputs, key=lambda item: (item["line"], item["path"])),
901        "occurrences": occurrences or [],
902        "calls": calls or [],
903    }

922def _matlab_code_part(line: str) -> str:
923    result: list[str] = []
924    in_string = False
925    quote = ""
926    index = 0
927    while index < len(line):
928        character = line[index]
929        if in_string:
930            result.append(character)
931            if character == quote:
932                if index + 1 < len(line) and line[index + 1] == quote:
933                    result.append(line[index + 1])
934                    index += 1
935                else:
936                    in_string = False
937        elif character == "'" and _is_matlab_transpose(result):
938            result.append(character)
939        elif character in {"'", '"'}:
940            in_string = True
941            quote = character
942            result.append(character)
943        elif character == "%":
944            break
945        else:
946            result.append(character)
947        index += 1
948    if in_string:
949        raise ValueError("unterminated MATLAB string")
950    return "".join(result)

953def _is_matlab_transpose(characters: list[str]) -> bool:
954    previous = next((character for character in reversed(characters) if not character.isspace()), "")
955    return bool(previous and (previous.isalnum() or previous in "_)]}"))

958def _sha256(path: Path) -> str:
959    digest = hashlib.sha256()
960    with path.open("rb") as handle:
961        for chunk in iter(lambda: handle.read(1024 * 1024), b""):
962            digest.update(chunk)
963    return digest.hexdigest()

987def _line_excerpt(lines: list[str], start: int, end: int, limit: int = 220) -> str:
988    if not lines or start < 1:
989        return "<source unavailable>"
990    selected = " ".join(line.strip() for line in lines[start - 1:min(end, len(lines))])
991    selected = re.sub(r"[\x00-\x1f\x7f]", " ", selected)
992    selected = re.sub(r"\s+", " ", selected)
993    return selected if len(selected) <= limit else selected[: limit - 3] + "..."
