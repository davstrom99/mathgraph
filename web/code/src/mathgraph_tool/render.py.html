37def render_static_site(
38    output_dir: str | Path = DEFAULT_WEB_DIR,
39    graph: GraphSpec | None = None,
40    *,
41    compile_tex: bool = True,
42) -> list[Path]:
43    """Render the graph viewer and return generated paths."""
44    repo_root = find_repo_root()
45    output_path = Path(output_dir)
46    if not output_path.is_absolute():
47        output_path = repo_root / output_path
48    output_path.mkdir(parents=True, exist_ok=True)
49
50    graph = graph or load_graph(repo_root=repo_root)
51    _copy_referenced_files(graph, repo_root, output_path)
52    title_images, title_report = _render_node_title_images(graph, output_path)
53    tex_report = _render_tex_files(graph, repo_root, output_path, compile_tex=compile_tex)
54    _render_code_files(graph, repo_root, output_path)
55    static_assets = _copy_static_assets(output_path)
56    files: dict[Path, str] = {
57        output_path / "graph.json": json.dumps(_graph_payload(graph, repo_root, title_images), indent=2),
58        output_path / "index.html": _index_html(),
59        output_path / "style.css": _style_css(),
60        output_path / "app.js": _app_js(),
61        output_path / "render-report.json": json.dumps(
62            {"node_titles": title_report, "tex": tex_report}, indent=2
63        ),
64    }
65    for path, text in files.items():
66        path.write_text(text, encoding="utf-8")
67    return list(files) + static_assets

70def _copy_referenced_files(graph: GraphSpec, repo_root: Path, output_path: Path) -> None:
71    refs_root = output_path / "refs"
72    if refs_root.exists():
73        shutil.rmtree(refs_root)
74    for relative_path in _referenced_paths(graph):
75        source = repo_root / relative_path
76        if not source.exists() or not source.is_file():
77            continue
78        destination_relative = relative_path
79        if relative_path.suffix in CODE_EXTENSIONS:
80            destination_relative = relative_path.with_suffix(relative_path.suffix + ".source.txt")
81        destination = refs_root / destination_relative
82        destination.parent.mkdir(parents=True, exist_ok=True)
83        shutil.copy2(source, destination)

86def _render_tex_files(
87    graph: GraphSpec,
88    repo_root: Path,
89    output_path: Path,
90    *,
91    compile_tex: bool,
92) -> list[dict]:
93    tex_paths = {Path(ref.file) for _, ref in graph.all_tex_refs()}
94    derivation_paths = {Path(edge.tex.file) for edge in graph.edges}
95    if graph.project.tex_root:
96        tex_paths.add(Path(graph.project.tex_root))
97
98    compile_results: dict[Path, dict] = {}
99    if compile_tex and shutil.which("pdflatex"):
100        jobs = []
101        with ThreadPoolExecutor(max_workers=min(4, max(1, len(derivation_paths)))) as executor:
102            for relative_path in sorted(derivation_paths):
103                source = repo_root / relative_path
104                rendered_pdf = output_path / _tex_render_path(relative_path).with_suffix(".pdf")
105                if source.exists() and source.is_file():
106                    jobs.append((relative_path, executor.submit(_compile_tex_pdf, source, rendered_pdf, repo_root)))
107            for relative_path, future in jobs:
108                compile_results[relative_path] = future.result()
109
110    report: list[dict] = []
111    for relative_path in sorted(tex_paths):
112        source = repo_root / relative_path
113        if not source.exists() or not source.is_file():
114            continue
115        rendered = output_path / _tex_render_path(relative_path)
116        rendered.parent.mkdir(parents=True, exist_ok=True)
117        style_href = Path(os.path.relpath(output_path / "style.css", rendered.parent)).as_posix()
118        compilation = compile_results.get(relative_path)
119        pdf_name = rendered.with_suffix(".pdf").name if compilation and compilation["status"] in {"compiled", "cached"} else None
120        rendered.write_text(
121            _render_tex_document(
122                relative_path.as_posix(),
123                style_href,
124                source.read_text(encoding="utf-8-sig"),
125                pdf_name=pdf_name,
126                render_error=compilation.get("error") if compilation else None,
127            ),
128            encoding="utf-8",
129        )
130        report.append(
131            {
132                "file": relative_path.as_posix(),
133                "kind": "derivation" if relative_path in derivation_paths else "document",
134                "status": compilation["status"] if compilation else "browser-rendered",
135                "error": compilation.get("error") if compilation else None,
136            }
137        )
138    return report

141def _render_node_title_images(graph: GraphSpec, output_path: Path) -> tuple[dict[str, dict], list[dict]]:
142    from matplotlib.backends.backend_svg import FigureCanvasSVG
143    from matplotlib.figure import Figure
144
145    image_dir = output_path / "math" / "nodes"
146    image_dir.mkdir(parents=True, exist_ok=True)
147    images: dict[str, dict] = {}
148    report: list[dict] = []
149    for node in graph.nodes:
150        label = node.display_label or node.title
151        filename = re.sub(r"[^A-Za-z0-9_.-]+", "_", node.id) + ".svg"
152        destination = image_dir / filename
153        expression = _mathtext_label(label)
154        try:
155            figure = Figure(figsize=(3.2, 0.55), dpi=120)
156            FigureCanvasSVG(figure)
157            figure.patch.set_alpha(0)
158            figure.text(0.01, 0.5, expression, fontsize=14, va="center", color="#18212b")
159            figure.savefig(destination, format="svg", transparent=True, bbox_inches="tight", pad_inches=0.02)
160            title_width, title_height = _parse_svg_dimensions(destination)
161            box_width, box_height = _node_box_dimensions(title_width, title_height)
162            images[node.id] = {
163                "path": (Path("math") / "nodes" / filename).as_posix(),
164                "titleWidth": title_width,
165                "titleHeight": title_height,
166                "boxWidth": box_width,
167                "boxHeight": box_height,
168            }
169            report.append(
170                {
171                    "node": node.id,
172                    "status": "rendered",
173                    "error": None,
174                    "titleWidth": title_width,
175                    "titleHeight": title_height,
176                }
177            )
178        except Exception as exc:
179            title_width, title_height = _fallback_title_dimensions(label)
180            box_width, box_height = _node_box_dimensions(title_width, title_height)
181            images[node.id] = {
182                "path": None,
183                "titleWidth": title_width,
184                "titleHeight": title_height,
185                "boxWidth": box_width,
186                "boxHeight": box_height,
187            }
188            report.append(
189                {
190                    "node": node.id,
191                    "status": "text-fallback",
192                    "error": str(exc),
193                    "titleWidth": title_width,
194                    "titleHeight": title_height,
195                }
196            )
197    return images, report

212def _parse_svg_dimensions(path: Path) -> tuple[float, float]:
213    root = ElementTree.fromstring(path.read_text(encoding="utf-8"))
214    width = _parse_svg_length(root.get("width"))
215    height = _parse_svg_length(root.get("height"))
216    if width is not None and height is not None:
217        return width, height
218    view_box = root.get("viewBox", "").strip().split()
219    if len(view_box) == 4:
220        return float(view_box[2]), float(view_box[3])
221    raise ValueError(f"could not determine SVG dimensions for {path}")

224def _parse_svg_length(value: str | None) -> float | None:
225    if not value:
226        return None
227    match = re.search(r"([0-9]+(?:\.[0-9]+)?)", value)
228    if not match:
229        return None
230    return float(match.group(1))

233def _fallback_title_dimensions(label: str) -> tuple[float, float]:
234    plain = re.sub(r"\\[()[\]]", "", label).strip()
235    width = max(96.0, min(420.0, 24.0 + len(plain) * 7.4))
236    return width, 24.0

246def _mathtext_label(label: str) -> str:
247    value = label.strip()
248    if value.startswith(r"\(") and value.endswith(r"\)"):
249        return f"${value[2:-2]}$"
250    if value.startswith(r"\[") and value.endswith(r"\]"):
251        return f"${value[2:-2]}$"
252    return value

255def _compile_tex_pdf(source: Path, destination: Path, repo_root: Path) -> dict:
256    metadata = destination.with_suffix(".pdf.meta.json")
257    signature = {
258        "version": TEX_RENDER_VERSION,
259        "source_mtime_ns": source.stat().st_mtime_ns,
260        "source_size": source.stat().st_size,
261    }
262    if destination.exists() and metadata.exists():
263        try:
264            if json.loads(metadata.read_text(encoding="utf-8")) == signature:
265                return {"status": "cached", "error": None}
266        except (json.JSONDecodeError, OSError):
267            pass
268
269    destination.parent.mkdir(parents=True, exist_ok=True)
270    content = source.read_text(encoding="utf-8-sig")
271    if r"\documentclass" not in content:
272        content = (
273            "\\documentclass{article}\n"
274            "\\usepackage{amsmath,amssymb,bm,mathtools}\n"
275            "\\usepackage[margin=0.65in]{geometry}\n"
276            "\\pagestyle{empty}\n"
277            "\\begin{document}\n"
278            f"{content}\n"
279            "\\end{document}\n"
280        )
281    try:
282        with tempfile.TemporaryDirectory(prefix="mathgraph-tex-") as temp_name:
283            temp = Path(temp_name)
284            tex_path = temp / "derivation.tex"
285            tex_path.write_text(content, encoding="utf-8")
286            environment = os.environ.copy()
287            existing_inputs = environment.get("TEXINPUTS", "")
288            environment["TEXINPUTS"] = os.pathsep.join(
289                [str(source.parent), str(repo_root), existing_inputs]
290            )
291            completed = subprocess.run(
292                [
293                    shutil.which("pdflatex") or "pdflatex",
294                    "-interaction=nonstopmode",
295                    "-halt-on-error",
296                    f"-output-directory={temp}",
297                    str(tex_path),
298                ],
299                cwd=source.parent,
300                env=environment,
301                capture_output=True,
302                text=True,
303                timeout=45,
304                check=False,
305            )
306            pdf = temp / "derivation.pdf"
307            if completed.returncode != 0 or not pdf.exists():
308                output = (completed.stdout + "\n" + completed.stderr).strip()
309                return {"status": "browser-fallback", "error": output[-1200:] or "pdflatex failed"}
310            shutil.copy2(pdf, destination)
311            metadata.write_text(json.dumps(signature, sort_keys=True), encoding="utf-8")
312            return {"status": "compiled", "error": None}
313    except (OSError, subprocess.SubprocessError) as exc:
314        return {"status": "browser-fallback", "error": str(exc)}

346def _graph_payload(graph: GraphSpec, repo_root: Path, title_images: dict[str, dict] | None = None) -> dict:
347    incoming: dict[str, list[Edge]] = {node.id: [] for node in graph.nodes}
348    outgoing: dict[str, list[Edge]] = {node.id: [] for node in graph.nodes}
349    for edge in graph.edges:
350        outgoing[edge.from_].append(edge)
351        incoming[edge.to].append(edge)
352    cluster_memberships, clusters = _cluster_memberships(graph)
353
354    return {
355        "project": graph.project.model_dump(mode="json"),
356        "nodes": [
357            _node_payload(
358                node,
359                incoming[node.id],
360                outgoing[node.id],
361                repo_root,
362                (title_images or {}).get(node.id),
363                cluster_memberships.get(node.id),
364            )
365            for node in graph.nodes
366        ],
367        "edges": [_edge_payload(edge, repo_root) for edge in graph.edges],
368        "clusters": clusters,
369    }

406def _cluster_memberships(graph: GraphSpec) -> tuple[dict[str, dict], list[dict]]:
407    graph_nx = nx.DiGraph()
408    graph_nx.add_nodes_from(node.id for node in graph.nodes)
409    graph_nx.add_edges_from((edge.from_, edge.to) for edge in graph.edges)
410
411    memberships: dict[str, dict] = {}
412    clusters: list[dict] = []
413    nontrivial = [
414        sorted(component)
415        for component in nx.strongly_connected_components(graph_nx)
416        if len(component) > 1
417    ]
418    nontrivial.sort(key=lambda members: (-len(members), members))
419    for index, members in enumerate(nontrivial, start=1):
420        cluster_id = f"scc.{index}"
421        summary = {
422            "id": cluster_id,
423            "size": len(members),
424            "members": members,
425            "label": f"SCC ({len(members)} nodes)",
426        }
427        clusters.append(summary)
428        for node_id in members:
429            memberships[node_id] = summary
430    return memberships, clusters

517def _symbol_range(path: Path, symbol: str | None) -> tuple[int | None, int | None]:
518    if symbol is None or not path.exists() or path.suffix != ".py":
519        return None, None
520    try:
521        tree = ast.parse(path.read_text(encoding="utf-8"), filename=str(path))
522    except SyntaxError:
523        return None, None
524    for item in tree.body:
525        if isinstance(item, ast.FunctionDef | ast.AsyncFunctionDef | ast.ClassDef) and item.name == symbol:
526            return item.lineno, getattr(item, "end_lineno", item.lineno)
527    return None, None

530def _symbol_ranges_for_file(text: str, path: str) -> dict[str, tuple[int, int]]:
531    if not path.endswith(".py"):
532        return {}
533    try:
534        tree = ast.parse(text)
535    except SyntaxError:
536        return {}
537    ranges: dict[str, tuple[int, int]] = {}
538    for item in tree.body:
539        if isinstance(item, ast.FunctionDef | ast.AsyncFunctionDef | ast.ClassDef):
540            ranges[item.name] = (item.lineno, getattr(item, "end_lineno", item.lineno))
541    return ranges

544def _render_code_document(title: str, style_href: str, text: str) -> str:
545    ranges = _symbol_ranges_for_file(text, title)
546    line_to_symbol = {start: symbol for symbol, (start, _) in ranges.items()}
547    highlighted_lines = {
548        line
549        for start, end in ranges.values()
550        for line in range(start, end + 1)
551    }
552    rendered_lines = []
553    for number, line in enumerate(text.splitlines(), start=1):
554        anchors = ""
555        if number in line_to_symbol:
556            anchors = f'<a class="source-anchor" id="{escape(line_to_symbol[number])}"></a>'
557        classes = "source-line highlighted" if number in highlighted_lines else "source-line"
558        rendered_lines.append(
559            f'{anchors}<span id="L{number}" class="{classes}">'
560            f'<span class="line-number">{number}</span>'
561            f'<code>{escape(line)}</code>'
562            f"</span>"
563        )
564    body = "\n".join(rendered_lines)
565    return f"""<!doctype html>
566<html lang="en">
567  <head>
568    <meta charset="utf-8">
569    <meta name="viewport" content="width=device-width, initial-scale=1">
570    <title>{escape(title)}</title>
571    <link rel="stylesheet" href="{escape(style_href)}">
572  </head>
573  <body class="code-page">
574    <header class="code-header">
575      <h1>{escape(title)}</h1>
576    </header>
577    <pre class="source-view">{body}</pre>
578  </body>
579</html>
580"""

583def _render_tex_document(
584    title: str,
585    style_href: str,
586    text: str,
587    *,
588    pdf_name: str | None = None,
589    render_error: str | None = None,
590) -> str:
591    if pdf_name:
592        return f"""<!doctype html>
593<html lang="en">
594  <head>
595    <meta charset="utf-8">
596    <meta name="viewport" content="width=device-width, initial-scale=1">
597    <title>{escape(title)}</title>
598    <link rel="stylesheet" href="{escape(style_href)}">
599    <style>

601      .derivation-frame {{ display: block; width: 100%; height: 100vh; border: 0; }}
602      .pdf-link {{ position: fixed; right: 12px; top: 10px; z-index: 2; background: #fff; padding: 6px 9px; border: 1px solid #d9e0e8; border-radius: 4px; }}
603    </style>
604  </head>

655def _tex_to_html(text: str) -> str:
656    text = re.sub(r"\\documentclass(?:\[[^\]]*\])?\{[^}]+\}", "", text)
657    text = re.sub(r"\\usepackage(?:\[[^\]]*\])?\{[^}]+\}", "", text)
658    text = re.sub(r"\\title\{([^}]*)\}", r"<h1>\1</h1>", text)
659    text = re.sub(r"\\author\{([^}]*)\}", r"<p>\1</p>", text)
660    text = re.sub(r"\\date\{[^}]*\}", "", text)
661    text = text.replace(r"\begin{document}", "").replace(r"\end{document}", "")
662    text = text.replace(r"\maketitle", "")
663    text = re.sub(r"\\input\{([^}]+)\}", r"\n\nIncluded file: \1\n\n", text)
664    text = re.sub(r"\\section\*?\{([^}]*)\}", r"\n\n<h2>\1</h2>\n\n", text)
665    text = re.sub(r"\\subsection\*?\{([^}]*)\}", r"\n\n<h3>\1</h3>\n\n", text)
666    labels = re.findall(r"\\label\{([^}]*)\}", text)
667    text = re.sub(r"\\label\{[^}]*\}", "", text)
668    text = re.sub(r"\\par(?:\s+|(?=[A-Z]))", "\n\n", text)
669
670    math_blocks: list[str] = []
671
672    def stash_math(value: str) -> str:
673        math_blocks.append(value)
674        return f"\n\n@@MATH{len(math_blocks) - 1}@@\n\n"
675
676    environment_pattern = re.compile(
677        r"\\begin\{(align\*?|equation\*?|gather\*?|multline\*?|displaymath)\}([\s\S]*?)\\end\{\1\}"
678    )
679
680    def replace_environment(match: re.Match) -> str:
681        environment = match.group(1).rstrip("*")
682        content = match.group(2).strip()
683        if environment in {"equation", "displaymath"}:
684            rendered = f"\\[{content}\\]"
685        elif environment == "gather":
686            rendered = f"\\[\\begin{{gathered}}{content}\\end{{gathered}}\\]"
687        else:
688            rendered = f"\\[\\begin{{aligned}}{content}\\end{{aligned}}\\]"
689        return stash_math(rendered)
690
691    text = environment_pattern.sub(replace_environment, text)
692    text = re.sub(r"\\\[[\s\S]*?\\\]", lambda match: stash_math(match.group(0)), text)
693
694    parts = re.split(r"(@@MATH\d+@@|<h[1-3]>.*?</h[1-3]>)", text)
695    html_parts: list[str] = []
696    html_parts.extend(f'<a class="anchor" id="{escape(label)}"></a>' for label in labels)
697    for part in parts:
698        stripped = part.strip()
699        if not stripped:
700            continue
701        if stripped.startswith("<h"):
702            html_parts.append(stripped)
703        elif re.fullmatch(r"@@MATH\d+@@", stripped):
704            index = int(stripped[6:-2])
705            html_parts.append(f'<div class="math">{escape(math_blocks[index])}</div>')
706        else:
707            paragraphs = [p.strip() for p in stripped.split("\n\n") if p.strip()]
708            for paragraph in paragraphs:
709                html_parts.append(f"<p>{_render_inline_latex(paragraph)}</p>")
710    return "\n".join(html_parts)

713def _render_inline_latex(text: str) -> str:
714    placeholders: list[str] = []
715
716    def stash(pattern: str, template) -> None:
717        nonlocal text
718
719        def replace(match: re.Match) -> str:
720            placeholders.append(template(match))
721            return f"@@HTML{len(placeholders) - 1}@@"
722
723        text = re.sub(pattern, replace, text)
724
725    stash(r"\\texttt\{([^}]*)\}", lambda match: f"<code>{_latex_text_arg(match.group(1))}</code>")
726    stash(r"\\textbf\{([^}]*)\}", lambda match: f"<strong>{_latex_text_arg(match.group(1))}</strong>")
727    stash(r"\\emph\{([^}]*)\}", lambda match: f"<em>{_latex_text_arg(match.group(1))}</em>")
728
729    escaped = escape(text)
730    for index, value in enumerate(placeholders):
731        escaped = escaped.replace(f"@@HTML{index}@@", value)
732    return escaped

1615function zoomBy(factor, clientX = null, clientY = null) {
1616  const oldScale = transform.scale;
1617  const newScale = oldScale * factor;
1618  if (!Number.isFinite(newScale) || newScale <= 0) return;
1619  if (newScale === oldScale) return;
1620
1621  const rect = graphView.getBoundingClientRect();
1622  const scale = screenScale();
1623  const screen = clientX === null || clientY === null
1624    ? { x: rect.left + rect.width / 2, y: rect.top + rect.height / 2 }
1625    : { x: clientX, y: clientY };
1626  const localX = screen.x - rect.left;
1627  const localY = screen.y - rect.top;
1628  const anchor = screenToGraphPoint(screen.x, screen.y);
1629
1630  transform.scale = newScale;
1631  transform.x = localX / scale.x - anchor.x * newScale;
1632  transform.y = localY / scale.y - anchor.y * newScale;
1633  scheduleGraphTransform();
1634}
