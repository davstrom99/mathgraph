12SCAFFOLD_PATHS = [
13    Path("mathgraph/graph.yaml"),
14    Path("mathgraph/schema.yaml"),
15    Path("mathgraph/INIT_CHECKLIST.md"),
16    Path("mathgraph/INIT_REPORT.md"),

27@dataclass(frozen=True)
28class InitResult:
29    planned: list[Path]
30    written: list[Path]
31    existing: list[Path]
32    checklist_count: int
33    dry_run: bool = False
34    refused: bool = False
35    blocked_count: int = 0
36    review_mode: str = "source"
37
38    @property
39    def ok(self) -> bool:
40        return not self.refused and (self.review_mode == "full" or self.blocked_count == 0)

43def initialize_mathgraph(
44    repo_root: Path,
45    *,
46    project_id: str | None = None,
47    title: str | None = None,
48    force: bool = False,
49    dry_run: bool = False,
50    include_extensions: set[str] | None = None,
51    exclude_dirs: set[str] | None = None,
52    full_review: bool = False,
53    include_generated: bool = False,
54) -> InitResult:
55    """Create mathgraph scaffolding for a repo."""
56    root = repo_root.resolve()
57    planned = [root / path for path in SCAFFOLD_PATHS] + [root / AGENTS_PATH]
58    existing = [path for path in planned if path.exists()]
59    blocking_existing = [root / path for path in SCAFFOLD_PATHS if (root / path).exists()]
60    if blocking_existing and not force:
61        return InitResult(planned=planned, written=[], existing=blocking_existing, checklist_count=0, dry_run=dry_run, refused=True)
62
63    discovered = discover_project_files(root, include_extensions=include_extensions, exclude_dirs=exclude_dirs)
64    index = build_index(
65        root,
66        files=discovered,
67        full_review=full_review,
68        write=False,
69        include_extensions=include_extensions,
70        exclude_dirs=exclude_dirs,
71        include_generated=include_generated,
72    )
73    project_id = project_id or _default_project_id(root)
74    title = title or root.name.replace("_", " ").replace("-", " ").title()
75    contents = {
76        root / "mathgraph/graph.yaml": _graph_yaml(project_id, title),
77        root / "mathgraph/schema.yaml": _schema_yaml(),
78        root / "mathgraph/INIT_CHECKLIST.md": _checklist(title, discovered, index),
79        root / "mathgraph/INIT_REPORT.md": _init_report(title, index),
80        root / "mathgraph/INIT_DRAFT.yaml": serialize_initial_draft(index, root),
81        root / INDEX_PATH: serialize_index(index),
82        root / "mathgraph/paper/main.tex": _main_tex(title),
83    }
84
85    if dry_run:
86        return InitResult(
87            planned=planned,
88            written=[],
89            existing=existing,
90            checklist_count=_checklist_count(index),
91            dry_run=True,
92            blocked_count=index.blocked,
93            review_mode=index.review_mode,
94        )
95
96    written: list[Path] = []
97    for path, text in contents.items():
98        path.parent.mkdir(parents=True, exist_ok=True)
99        path.write_text(text, encoding="utf-8")
100        written.append(path)
101    agents_path = root / AGENTS_PATH
102    if _write_or_append_agents(agents_path):
103        written.append(agents_path)
104    return InitResult(
105        planned=planned,
106        written=written,
107        existing=existing,
108        checklist_count=_checklist_count(index),
109        blocked_count=index.blocked,
110        review_mode=index.review_mode,
111    )

114def format_init_result(result: InitResult) -> str:
115    if result.refused:
116        lines = ["Refusing to initialize because scaffold files already exist:"]
117        lines.extend(f"- {_display_path(path)}" for path in result.existing)
118        lines.append("")
119        lines.append("Use --force to overwrite or inspect the existing files first.")
120        return "\n".join(lines)
121
122    action = "Would create" if result.dry_run else "Created"
123    lines = [f"{action} mathgraph scaffold:"]
124    paths = result.planned if result.dry_run else result.written
125    lines.extend(f"- {_display_path(path)}" for path in paths)
126    lines.append("")
127    lines.append(f"Checklist entries: {result.checklist_count}")
128    lines.append(f"Review mode: {result.review_mode}")
129    lines.append(f"Blocked index records: {result.blocked_count}")
130    lines.append("")
131    lines.append("Next steps:")
132    lines.append("- If invoked from Codex as `/mathgraph init`, continue the full onboarding workflow now.")
133    if result.review_mode == "full":
134        lines.append("- Read every file in mathgraph/INIT_CHECKLIST.md completely and fill the review record.")
135    else:
136        lines.append("- Use mathgraph index dossier first; inspect every authoritative source record and resolve identities with draft-refs/find.")
137    if result.blocked_count and result.review_mode == "indexed":
138        lines.append("- Initialization is blocked: exclude unsupported files or explicitly invoke `/mathgraph init --full-review`.")
139    lines.append("- Write findings to mathgraph/INIT_REPORT.md, then construct the initial graph and TeX references.")
140    lines.append("- Run mathgraph check, mathgraph orphans, mathgraph coverage, pytest when available, and mathgraph render.")
141    return "\n".join(lines)

144def _display_path(path: Path) -> str:
145    try:
146        return path.resolve().relative_to(Path.cwd().resolve()).as_posix()
147    except ValueError:
148        return path.as_posix()

151def _default_project_id(root: Path) -> str:
152    value = root.name.lower().replace(" ", "_").replace("-", "_")
153    return "".join(character for character in value if character.isalnum() or character == "_") or "mathgraph_project"

235def _agents_md() -> str:
236    return f"""{AGENTS_SECTION_START}
237## Mathgraph-Centered Repository Rules
238
239This repository is mathgraph-centered.
240
241Before editing code:
242
2431. Work through `mathgraph/INIT_CHECKLIST.md` if the initial graph is still being constructed.
2442. For `/mathgraph init`, semantically review authoritative source by default, use `--include-generated` only when requested, and use complete file review only for explicit `--full-review`; never escalate automatically.
2453. Do not claim initialization complete while `mathgraph index check` reports blocked, missing, or stale evidence.
2464. Identify the relevant mathgraph node or nodes.
2475. Update the graph before or together with code changes.
2486. Every nontrivial function must attach to a graph node.
2497. Every mathematical edge must have a TeX derivation reference.
2508. Each variable node contains one variable only and uses its variable-and-domain equation as the title.
2519. Every node must reference every source line where its mathematical object is used, using exact `path` plus `line` code entries.
25210. Prefer a meaningful one-line equation as a node title; otherwise use a concise description. Equation titles must render to stable local SVG assets.
25311. Store every distinct derivation in its own standalone-compilable TeX file. Multiple edges may share one file and label when they use the same argument.
25412. After `mathgraph render`, inspect `web/render-report.json` and fix title or derivation fallbacks when local renderers are available.
25513. Run `mathgraph check`, `mathgraph orphans`, and relevant tests after changes.
256{AGENTS_SECTION_END}
257"""

260def _write_or_append_agents(path: Path) -> bool:
261    section = _agents_md()
262    if not path.exists():
263        path.write_text(f"# Repository Rules for Codex\n\n{section}", encoding="utf-8")
264        return True
265
266    existing = path.read_text(encoding="utf-8")
267    if AGENTS_SECTION_START in existing:
268        return False
269
270    separator = "\n\n" if existing.endswith("\n") else "\n\n"
271    path.write_text(f"{existing}{separator}{section}", encoding="utf-8")
272    return True

275def _checklist(title: str, files: list[DiscoveredFile], index: IndexBuildResult) -> str:
276    records = {
277        record["path"]: record
278        for record in index.records
279        if record.get("record_type") == "file"
280    }
281    full_review = index.review_mode == "full"
282    lines = [
283        f"# Mathgraph Initialization Checklist: {title}",
284        "",
285        "Codex instruction for `/mathgraph init`: this is a work queue, not a passive checklist.",
286        "",
287        "Complete every item before claiming initialization is done:",
288        "",
289        f"Review mode: `{index.review_mode}`.",
290        "",
291        "1. Run `mathgraph index check` and `mathgraph index dossier`.",
292        (
293            "2. Read every eligible file completely; the index remains the traceability source."
294            if full_review
295            else "2. Inspect every authoritative source record; use `mathgraph index show <path>` only for needed detail and aggregate non-authoritative families unless this mode explicitly includes them."
296        ),
297        "3. Mark each authoritative file reviewed only after semantic review; verify each generated family by its aggregate hash record.",
298        "4. Summarize cross-file findings in `mathgraph/INIT_REPORT.md`.",
299        "5. Create or update `mathgraph/graph.yaml` and `mathgraph/paper/...` so the project has a usable initial graph.",
300        "6. Give each variable its own node titled by its variable-and-domain equation; prefer one-line equations for other node titles when meaningful.",
301        "7. Use `mathgraph draft-refs <symbol-or-node>` to mechanically generate exact role-classified `path` plus `line` references, then resolve ambiguous identities.",
302        "8. Put each distinct derivation in its own standalone TeX file; reuse the same file and label across edges that share the argument.",
303        "9. Run `mathgraph check`, `mathgraph orphans`, `mathgraph coverage`, `pytest` when available, and `mathgraph render`.",
304        "",
305        "Do not edit project source code during initialization unless the user explicitly asks for implementation changes.",
306        "Do not claim initialization complete while `mathgraph index check` reports blocked, missing, or stale files.",
307        "",
308    ]
309    if not files:
310        lines.append("No eligible project files were found.")
311        lines.append("")
312        return "\n".join(lines)
313
314    for item in files:
315        record = records.get(item.path.as_posix())
316        if record is None:
317            continue
318        details = (
319            f"parser={record['parser']}; status={record['status']}; "
320            f"lines={record['line_count']}; regions={len(record['regions'])}; symbols={len(record['symbols'])}"
321        )
322        lines.append(f"- [ ] `{item.path.as_posix()}` - {details}")
323    families = [record for record in index.records if record.get("record_type") == "family"]
324    if families:
325        lines.extend(["", "## Aggregate evidence families", ""])
326        for record in families:
327            lines.append(
328                f"- [ ] `{record['family']}` - {record['evidence_class']}; "
329                f"{record['file_count']} files; {record['size_bytes']} bytes; hashes verified"
330            )
331    lines.extend(["", "Record cross-file identities, graph decisions, use-site coverage, and unresolved questions in `mathgraph/INIT_REPORT.md`.", ""])
332    return "\n".join(lines)

335def _init_report(title: str, index: IndexBuildResult) -> str:
336    families = [record for record in index.records if record.get("record_type") == "family"]
337    authoritative_count = sum(
338        record.get("evidence_class") == "authoritative"
339        for record in index.records
340        if record.get("record_type") == "file"
341    )
342    family_lines = "\n".join(
343        f"- `{record['family']}`: {record['file_count']} files, hashes verified ({record['evidence_class']})"
344        for record in families
345    ) or "- None"
346    configured_excludes = list((index.classification or {}).get("exclude", ()))
347    exclusion_lines = "\n".join(f"- `{pattern}`" for pattern in configured_excludes) or "- Default generated/cache/rendered/output and binary classifications apply."
348    identity_locations: dict[str, int] = {}
349    for record in index.records:
350        if record.get("record_type") == "file":
351            for symbol in record.get("symbols", []):
352                identity_locations[symbol["name"]] = identity_locations.get(symbol["name"], 0) + 1
353    unresolved = "\n".join(
354        f"- `{name}`: {count} candidate definitions"
355        for name, count in sorted(identity_locations.items())
356        if count > 1
357    ) or "- None detected mechanically."
358    return f"""# Mathgraph Initialization Report: {title}
359
360This report is filled by Codex during `/mathgraph init`.
361
362- Review mode: `{index.review_mode}`
363- Evidence index: `mathgraph/INIT_INDEX.jsonl`

365

370{exclusion_lines}

372{family_lines}

378- Unreviewed files:

437def _checklist_count(index: IndexBuildResult) -> int:
438    return sum(1 for record in index.records if record.get("record_type") in {"file", "family"})
