121@dataclass(frozen=True)
122class DiscoveredFile:
123    path: Path
124    category: str
125    evidence_class: str = "authoritative"
126    family: str | None = None
127
128
129@dataclass(frozen=True)
130class DiscoveryConfig:
131    include: tuple[str, ...] = ()
132    exclude: tuple[str, ...] = ()
133    authoritative: tuple[str, ...] = ()
134    generated: tuple[str, ...] = ()
135

137def load_discovery_config(root: Path) -> DiscoveryConfig:
138    """Load optional source-classification rules from pyproject.toml."""
139    path = root.resolve() / "pyproject.toml"
140    if not path.exists():
141        return DiscoveryConfig()
142    try:
143        data = tomllib.loads(path.read_text(encoding="utf-8"))
144    except (OSError, tomllib.TOMLDecodeError):
145        return DiscoveryConfig()
146    values = data.get("tool", {}).get("mathgraph", {}).get("init", {})
147    if not isinstance(values, dict):
148        return DiscoveryConfig()
149    return DiscoveryConfig(
150        include=_string_tuple(values.get("include")),
151        exclude=_string_tuple(values.get("exclude")),
152        authoritative=_string_tuple(values.get("authoritative")),
153        generated=_string_tuple(values.get("generated")),
154    )

157def discover_project_files(
158    root: Path,
159    *,
160    include_extensions: set[str] | None = None,
161    exclude_dirs: set[str] | None = None,
162) -> list[DiscoveredFile]:
163    """Return eligible files for Codex-led graph initialization."""
164    root = root.resolve()
165    includes = {_normalize_extension(value) for value in include_extensions} if include_extensions else (CODE_EXTENSIONS | DOCUMENT_EXTENSIONS)
166    excluded = EXCLUDED_DIR_NAMES
167    config = load_discovery_config(root)
168    explicit_excludes = tuple(config.exclude) + tuple(f"**/{name}/**" for name in sorted(exclude_dirs or set()))
169    discovered: list[DiscoveredFile] = []
170
171    for path in _walk_files(root, excluded):
172        relative = path.relative_to(root)
173        if _is_excluded(relative, excluded):
174            continue
175        suffix = _normalized_suffix(path)
176        evidence_class, family = _classify(relative, config, explicit_excludes)
177        explicitly_included = _matches_any(relative.as_posix(), config.include)
178        if suffix in BINARY_EXTENSIONS and not explicitly_included and not _matches_any(relative.as_posix(), config.authoritative):
179            evidence_class, family = "binary", "binary artifacts"
180        if suffix not in includes and evidence_class == "authoritative" and not explicitly_included:
181            continue
182        category = "code" if suffix in CODE_EXTENSIONS else "document"
183        if suffix not in includes:
184            category = "binary"
185        discovered.append(DiscoveredFile(relative, category, evidence_class, family))
186    return discovered

189def _walk_files(root: Path, excluded_dirs: set[str]) -> list[Path]:
190    files: list[Path] = []
191    for current, dirs, filenames in os.walk(root, topdown=True, followlinks=False, onerror=lambda _: None):
192        dirs[:] = sorted(directory for directory in dirs if directory not in excluded_dirs)
193        current_path = Path(current)
194        for filename in sorted(filenames):
195            files.append(current_path / filename)
196    return files

199def _is_excluded(relative: Path, excluded_dirs: set[str]) -> bool:
200    if any(part in excluded_dirs for part in relative.parts[:-1]):
201        return True
202    name = relative.name
203    if name in EXCLUDED_FILE_NAMES:
204        return True
205    if name.startswith(".") and name not in {".env.example"}:
206        return True
207    return any(name.endswith(suffix) for suffix in EXCLUDED_FILE_SUFFIXES)

210def _normalized_suffix(path: Path) -> str:
211    name = path.name
212    for suffix in EXCLUDED_FILE_SUFFIXES:
213        if name.endswith(suffix):
214            return suffix
215    return path.suffix

218def _normalize_extension(value: str) -> str:
219    return value if value.startswith(".") else f".{value}"

222def _classify(relative: Path, config: DiscoveryConfig, explicit_excludes: tuple[str, ...]) -> tuple[str, str | None]:
223    value = relative.as_posix()
224    if _matches_any(value, config.authoritative) or _matches_any(value, config.include):
225        return "authoritative", None
226    if _matches_any(value, explicit_excludes):
227        pattern = next(pattern for pattern in explicit_excludes if _matches(value, pattern))
228        return "excluded", _family_name(pattern, value)
229    if _matches_any(value, config.generated):
230        pattern = next(pattern for pattern in config.generated if _matches(value, pattern))
231        return "generated", _family_name(pattern, value)
232    for evidence_class, patterns in DEFAULT_CLASSIFICATION_PATTERNS.items():
233        for pattern in patterns:
234            if _matches(value, pattern):
235                return evidence_class, _family_name(pattern, value)
236    return "authoritative", None

239def _matches_any(value: str, patterns: tuple[str, ...]) -> bool:
240    return any(_matches(value, pattern) for pattern in patterns)
241

243def _matches(value: str, pattern: str) -> bool:
244    normalized = pattern.replace("\\", "/").lstrip("./")
245    return fnmatch.fnmatch(value, normalized) or Path(value).match(normalized)

248def _family_name(pattern: str, value: str) -> str:
249    prefix = pattern.replace("\\", "/").split("*")[0].rstrip("/")
250    if prefix.startswith("**/") or not prefix:
251        parts = value.split("/")
252        if ".egg-info" in value:
253            return next(("/".join(parts[: index + 1]) for index, part in enumerate(parts) if part.endswith(".egg-info")), parts[0])
254        if "__pycache__" in parts:
255            index = parts.index("__pycache__")
256            return "/".join(parts[: index + 1])
257        return parts[0]
258    return prefix
