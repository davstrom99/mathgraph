8def simulate_gaussian_linear(
9    n: int,
10    d: int,
11    beta: np.ndarray,
12    sigma: float,
13    seed: int | None = None,
14) -> tuple[np.ndarray, np.ndarray]:
15    """Simulate data attached to graph node `model.gaussian_observation`."""
16    beta = np.asarray(beta, dtype=float)
17    if n <= 0:
18        raise ValueError("n must be positive")
19    if d <= 0:
20        raise ValueError("d must be positive")
21    if beta.shape != (d,):
22        raise ValueError(f"beta must have shape ({d},)")
23    if sigma <= 0:
24        raise ValueError("sigma must be positive")
25
26    rng = np.random.default_rng(seed)
27    X = rng.normal(size=(n, d))
28    noise = rng.normal(loc=0.0, scale=sigma, size=n)
29    y = X @ beta + noise
30    return X, y

33def simulate_student_t_linear(
34    n: int,
35    d: int,
36    beta: np.ndarray,
37    sigma: float,
38    nu: float,
39    seed: int | None = None,
40) -> tuple[np.ndarray, np.ndarray]:
41    """Simulate data attached to graph node `model.student_t_noise`."""
42    beta = np.asarray(beta, dtype=float)
43    if n <= 0:
44        raise ValueError("n must be positive")
45    if d <= 0:
46        raise ValueError("d must be positive")
47    if beta.shape != (d,):
48        raise ValueError(f"beta must have shape ({d},)")
49    if sigma <= 0:
50        raise ValueError("sigma must be positive")
51    if nu <= 0:
52        raise ValueError("nu must be positive")
53
54    rng = np.random.default_rng(seed)
55    X = rng.normal(size=(n, d))
56    noise = sigma * rng.standard_t(df=nu, size=n)
57    y = X @ beta + noise
58    return X, y
