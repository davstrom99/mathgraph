3from __future__ import annotations
4
5import numpy as np
6from scipy.special import gammaln

9def gaussian_log_likelihood(
10    X: np.ndarray,
11    y: np.ndarray,
12    beta: np.ndarray,
13    sigma: float,
14) -> float:
15    """Evaluate the log likelihood attached to graph node `likelihood.gaussian`."""
16    X = np.asarray(X, dtype=float)
17    y = np.asarray(y, dtype=float)
18    beta = np.asarray(beta, dtype=float)
19    _validate_regression_inputs(X, y, beta)
20    if sigma <= 0:
21        raise ValueError("sigma must be positive")
22
23    residual = y - X @ beta
24    n = y.shape[0]
25    return float(
26        -0.5 * n * np.log(2.0 * np.pi * sigma**2)
27        -0.5 * np.dot(residual, residual) / sigma**2
28    )
29
30
31def _validate_regression_inputs(X: np.ndarray, y: np.ndarray, beta: np.ndarray) -> None:
32    if X.ndim != 2:
33        raise ValueError("X must be a two-dimensional array")
34    if y.ndim != 1:
35        raise ValueError("y must be a one-dimensional array")
36    if beta.ndim != 1:
37        raise ValueError("beta must be a one-dimensional array")
38    if X.shape[0] != y.shape[0]:
39        raise ValueError("X and y must have the same number of rows")
40    if X.shape[1] != beta.shape[0]:
41        raise ValueError("X columns must match beta length")

44def student_t_log_likelihood(
45    X: np.ndarray,
46    y: np.ndarray,
47    beta: np.ndarray,
48    sigma: float,
49    nu: float,
50) -> float:
51    """Evaluate the log likelihood attached to graph node `likelihood.student_t`."""
52    X = np.asarray(X, dtype=float)
53    y = np.asarray(y, dtype=float)
54    beta = np.asarray(beta, dtype=float)
55    _validate_regression_inputs(X, y, beta)
56    if sigma <= 0:
57        raise ValueError("sigma must be positive")
58    if nu <= 0:
59        raise ValueError("nu must be positive")
60
61    residual = y - X @ beta
62    scaled_squared = (residual / sigma) ** 2
63    normalizer = (
64        gammaln((nu + 1.0) / 2.0)
65        - gammaln(nu / 2.0)
66        - 0.5 * np.log(nu * np.pi)
67        - np.log(sigma)
68    )
69    log_density = normalizer - 0.5 * (nu + 1.0) * np.log1p(scaled_squared / nu)
70    return float(np.sum(log_density))
