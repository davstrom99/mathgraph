11def gaussian_posterior(
12    X: np.ndarray,
13    y: np.ndarray,
14    sigma: float,
15    tau: float,
16) -> tuple[np.ndarray, np.ndarray]:
17    """Return (mu_n, Sigma_n) for graph node `posterior.gaussian_conjugate`."""
18    X = np.asarray(X, dtype=float)
19    y = np.asarray(y, dtype=float)
20    _validate_posterior_inputs(X, y, sigma, tau)
21
22    d = X.shape[1]
23    precision = (X.T @ X) / sigma**2 + np.eye(d) / tau**2
24    rhs = (X.T @ y) / sigma**2
25    sigma_n = np.linalg.inv(precision)
26    mu_n = np.linalg.solve(precision, rhs)
27    return mu_n, sigma_n

30def negative_log_posterior_gaussian(
31    beta: np.ndarray,
32    X: np.ndarray,
33    y: np.ndarray,
34    sigma: float,
35    tau: float,
36) -> float:
37    """Evaluate the objective for graph node `objective.gaussian_negative_log_posterior`."""
38    X = np.asarray(X, dtype=float)
39    y = np.asarray(y, dtype=float)
40    beta = np.asarray(beta, dtype=float)
41    _validate_posterior_inputs(X, y, sigma, tau, beta=beta)
42
43    residual = y - X @ beta
44    data_term = 0.5 * np.dot(residual, residual) / sigma**2
45    prior_term = 0.5 * np.dot(beta, beta) / tau**2
46    return float(data_term + prior_term)

49def beta_map_closed_form(
50    X: np.ndarray,
51    y: np.ndarray,
52    sigma: float,
53    tau: float,
54) -> np.ndarray:
55    """Return the closed-form MAP estimator for graph node `estimator.beta_map`."""
56    mu_n, _ = gaussian_posterior(X, y, sigma, tau)
57    return mu_n

60def _validate_posterior_inputs(
61    X: np.ndarray,
62    y: np.ndarray,
63    sigma: float,
64    tau: float,
65    beta: np.ndarray | None = None,
66) -> None:
67    if X.ndim != 2:
68        raise ValueError("X must be a two-dimensional array")
69    if y.ndim != 1:
70        raise ValueError("y must be a one-dimensional array")
71    if X.shape[0] != y.shape[0]:
72        raise ValueError("X and y must have the same number of rows")
73    if X.shape[1] == 0:
74        raise ValueError("X must have at least one column")
75    if sigma <= 0:
76        raise ValueError("sigma must be positive")
77    if tau <= 0:
78        raise ValueError("tau must be positive")
79    if beta is not None:
80        if beta.ndim != 1:
81            raise ValueError("beta must be a one-dimensional array")
82        if beta.shape[0] != X.shape[1]:
83            raise ValueError("beta length must match X columns")

86def student_t_map_numerical(
87    X: np.ndarray,
88    y: np.ndarray,
89    sigma: float,
90    tau: float,
91    nu: float,
92    initial_beta: np.ndarray | None = None,
93    maxiter: int = 1000,
94) -> np.ndarray:
95    """Return the numerical MAP approximation for graph node `approx.student_t_map`."""
96    X = np.asarray(X, dtype=float)
97    y = np.asarray(y, dtype=float)
98    if initial_beta is None:
99        initial_beta = beta_map_closed_form(X=X, y=y, sigma=sigma, tau=tau)
100    initial_beta = np.asarray(initial_beta, dtype=float)
101    _validate_posterior_inputs(X, y, sigma, tau, beta=initial_beta)
102    if nu <= 0:
103        raise ValueError("nu must be positive")
104    if maxiter <= 0:
105        raise ValueError("maxiter must be positive")
106
107    def objective(beta: np.ndarray) -> float:
108        prior_penalty = 0.5 * np.dot(beta, beta) / tau**2
109        return float(
110            -student_t_log_likelihood(X=X, y=y, beta=beta, sigma=sigma, nu=nu)
111            + prior_penalty
112        )
113
114    result = minimize(
115        objective,
116        initial_beta,
117        method="L-BFGS-B",
118        options={"maxiter": int(maxiter), "ftol": 1e-10, "gtol": 1e-6},
119    )
120    if not result.success:
121        raise RuntimeError(f"Student-t MAP optimization failed: {result.message}")
122    return np.asarray(result.x, dtype=float)
